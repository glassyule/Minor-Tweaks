// New less hardcoded v3 hitbox overriding thing! Wow!
// Animation and frame set here are not displayed, only their hitbox is used, specifically for things calling Player_CheckBreak
function Player_HandleHitboxOverride
	switch Stage.PlayerListPos
	
	case PLAYER_SONIC_A
		CheckEqual(Object[+2].Type, TypeName[Blue Shield])
		TempValue0 = CheckResult
		CheckEqual(Object[+2].ShieldType, SHIELD_INSTA)
		TempValue0 &= CheckResult
		if TempValue0 == true
			Player.HitboxOverrideAnim = ANI_HAMMER_DASH	// insta-shield hitbox dummy animation
			Player.HitboxOverrideFrame = 0
		else
			Player.HitboxOverrideAnim = -1
			Player.HitboxOverrideFrame = -1
		end if
		
		break
		
	
	case PLAYER_AMY_A
		if Player.Animation == ANI_HAMMER_JUMP
			// we're just using the hardcoded hitbox here, this is only necessary because the second half of the frames are missing hitboxes, good job Sonic Team!
			Player.HitboxOverrideAnim = ANI_HAMMER_JUMP
			Player.HitboxOverrideFrame = 0
		else
			Player.HitboxOverrideAnim = -1
			Player.HitboxOverrideFrame = -1
		end if
		
		break
	
	default
		Player.HitboxOverrideAnim = -1
		Player.HitboxOverrideFrame = -1
		break
		
	end switch
end function


// TempValue0-3 are hitbox left, right, top, and bottom respectively
function Player_CheckBreak
	
	if Player.HitboxOverrideAnim != -1
		TempValue4 = Player.Animation
		Player.Animation = Player.HitboxOverrideAnim
	end if
	
	if Player.HitboxOverrideFrame != -1
		TempValue5 = Player.Frame
		Player.Frame = Player.HitboxOverrideFrame
	end if
	
	PlayerObjectCollision(C_ENEMY, TempValue0, TempValue1, TempValue2, TempValue3)
	
	if Player.HitboxOverrideAnim != -1
		Player.Animation = TempValue4
	end if
	
	if Player.HitboxOverrideFrame != -1
		Player.Frame = TempValue5
	end if
	
end function

// Used by badniks, if the conditions are met, they get destroyed, otherwise you get hurt
function Player_BadnikBreak
	CheckEqual(Player.Animation, ANI_JUMPING)
	TempValue0  = CheckResult
#platform: Use_Origins
	CheckEqual(Player.Animation, ANI_HAMMER_JUMP)
	TempValue0 |= CheckResult
	CheckEqual(Player.Animation, ANI_HAMMER_DASH)
	TempValue0 |= CheckResult
#endplatform
	CheckEqual(Player.Animation, ANI_SPINDASH)
	TempValue0 |= CheckResult
#platform: Use_Origins
	CheckEqual(Player.Animation, ANI_GLIDING)
	TempValue0 |= CheckResult
	CheckEqual(Player.Animation, ANI_GLIDING_STOP)
	TempValue0 |= CheckResult
#endplatform

	ArrayPos0   = Player.EntityNo
	ArrayPos0  += 2
	CheckEqual(Object[ArrayPos0].Type, TypeName[Invincibility])
	TempValue0 |= CheckResult
	
	CheckEqual(Player.SuperState, SUPERSTATE_SUPER)
	TempValue0 |= CheckResult

	// you're invincible to badniks during the warping run
	if Warp.Timer > 0
		TempValue0 |= true
	end if
	
	if stage.playerListPos == PLAYER_TAILS_A
		if Player.JumpStrength != 0x38000						// Can't reflect while swimming
			if Player.State == Player_State_Fly					// If flying...
				if Object.YPos < Player.YPos					// Reflect only if it's above Tails
					TempValue0 |= true
				end if
			end if
		end if
	end if

#platform: Use_Origins
	if game.playMode == BOOT_PLAYMODE_MISSION
		if Stage.TimeEnabled == false
			TempValue0 = false
		end if
	end if
#endplatform

	if TempValue0 == true
		ResetObjectEntity(Object.EntityNo, Flower_TypeNo, 0, Object.XPos, Object.YPos)
		Object.DrawOrder = 4

		CreateTempObject(TypeName[Smoke Puff], 0, Object.XPos, Object.YPos)
		Object[TempObjectPos].DrawOrder = 4

		CreateTempObject(TypeName[Object Score], Score.Timer, Object.XPos, Object.YPos)
		Object[TempObjectPos].DrawOrder = 4

#platform: Use_Origins
		TempValue0 = false

		if Player.Animation == ANI_HAMMER_JUMP
			TempValue0 = true
		end if
		if Player.Animation == ANI_HAMMER_DASH
			TempValue0 = true
		end if
		
		CheckEqual(Stage.PlayerListPos, PLAYER_AMY_A)
		TempValue0 &= CheckResult
		if TempValue0 == true
			PlaySfx(31, false)	// SFX_G_HAMMERHIT
		else
			PlaySfx(8, false)	// SFX_G_DESTROY
		end if
#endplatform
#platform: Use_Standalone
		PlaySfx(8, false)	// SFX_G_DESTROY
#endplatform
		if Player.YVelocity > 0
			if Player.YVelocity >= Object.YPos
				Player.YVelocity -= 0x10000
			else
				if Options.AttractMode == false
					Player.YVelocity += Player.GravityStrength
					Player.YVelocity += Player.GravityStrength
				end if
				FlipSign(Player.YVelocity)
			end if
		else
			Player.YVelocity += 0x10000
		end if

		switch Score.Timer
		case 0
			Player.Score += 100
			break
		case 1
			Player.Score += 200
			break
		case 2
			Player.Score += 500
			break
		case 3
		case 4
		case 5
		case 6
		case 7
		case 8
		case 9
		case 10
		case 11
		case 12
		case 13
		case 14
			Player.Score += 1000
			break
		case 15
			Player.Score += 10000
			break
		end switch

		if Score.Timer < 15
			Score.Timer++
		end if

#platform: Use_Haptics
		HapticEffect(10, 0, 0, 0)
#endplatform

#platform: Use_Origins
		TempValue0 = 0
		game.callbackParam0 = 0
		CheckEqual(Player.State, Player_State_Roll)
		TempValue0  = CheckResult
		CheckEqual(Player.State, Player_State_RollJump)
		TempValue0 |= CheckResult
		if TempValue0 == true
			if Stage.PlayerListPos == PLAYER_SONIC
				game.callbackParam0 = KILL_ENEMY_ATTR_SPINDASH
				StageStatsUsabilityParam2 += 1
			end if
		end if
		StageStatsUsabilityParam1 += 1
		EngineCallback(NOTIFY_KILL_ENEMY)
		if game.playMode == BOOT_PLAYMODE_MISSION
			// Set during "M097 - Mercy"
			if game.missionFunctionNo == RULE_MERCY_BADNIKS
				game.forceKillPlayer = true
			end if
		end if
#endplatform
	else
		if Player.InvincibleTimer == 0
			Player.State = Player_State_GotHit

#platform: Use_Origins
			if game.playMode == BOOT_PLAYMODE_MISSION
				if game.missionFunctionNo == RULE_MERCY_BADNIKS
					game.missionValue = 1
				end if
			end if
#endplatform

			if Player.XPos > Object.XPos
				Player.Speed = 0x20000
			else
				Player.Speed = -0x20000
			end if
		end if
	end if
end function

// Do you really need an explanation?
function Player_Hit
	ArrayPos0  = Player.EntityNo
	ArrayPos0 += 2
	if Object[ArrayPos0].Type != TypeName[Invincibility]
		if Object[ArrayPos0].Type == TypeName[Blue Shield]
			CheckEqual(Object[ArrayPos0].ShieldType, SHIELD_INSTA)
		else
			CheckResult = false
		end if
		
		if CheckResult == false
			if Player.InvincibleTimer == 0
				Player.State = Player_State_GotHit
				if Player.XPos > Object.XPos
					Player.Speed =  0x20000
				else
					Player.Speed = -0x20000
				end if
			end if
		end if
	end if
end function

function Player_FireHit

function Player_ProjectileHit
	ArrayPos0  = Player.EntityNo
	ArrayPos0 += 2
	
	switch stage.playerListPos
	case PLAYER_SONIC_A
		CheckEqual(Object[ArrayPos0].Type, TypeName[Blue Shield])
		TempValue0 = CheckResult
		CheckEqual(Object[ArrayPos0].ShieldType, SHIELD_INSTA)	// Reflect it if Sonic is insta-shielding
		TempValue0 &= CheckResult
		
		break
	
	case PLAYER_TAILS_A
		TempValue0 = false
		
		if Player.JumpStrength != 0x38000						// Can't reflect while swimming
			if Player.State == Player_State_Fly					// If flying...
				if Object.YPos < Player.YPos					// Reflect only if it's above Tails
					TempValue0 = true
				end if
			end if
		end if
		
		break
	
	case PLAYER_KNUCKLES_A
		CheckEqual(Player.State, Player_State_GlideDrop)		// Reflect it if Knuckles is glide dropping...
		TempValue0 = CheckResult
		CheckEqual(Player.Gravity, GRAVITY_GROUND)				// ...but only on the ground
		TempValue0 &= CheckResult
		
		CheckEqual(Player.State, Player_State_GlideLeft)		// Also reflect it if Knuckles is gliding left...
		TempValue0 |= CheckResult
		CheckEqual(Player.State, Player_State_GlideRight)		// ...or gliding right...
		TempValue0 |= CheckResult
		CheckEqual(Player.State, Player_State_GlideLeftNoGrip)	// ...or gliding left without grip...
		TempValue0 |= CheckResult
		CheckEqual(Player.State, Player_State_GlideRightNoGrip)	// ...or gliding right without grip...
		TempValue0 |= CheckResult
		CheckEqual(Player.State, Player_State_GlideSlide)		// ...or sliding after a glide
		TempValue0 |= CheckResult
		
		break
	
	default
		TempValue0 = false	// Amy cannot reflect projectiles, so sad
		break
	end switch
	
	CheckGreater(Object[ArrayPos0].PropertyValue, ACTIVE_SHIELD)	// Also reflect it if the player has an elemental shield
	TempValue0 |= CheckResult
	
	if TempValue0 == true
		TempValue0 = Player.XPos
		TempValue0 -= Object.XPos

		TempValue1 = Player.YPos
		TempValue1 -= Object.YPos

		ATan2(TempValue2, TempValue0, TempValue1)
		Sin256(TempValue0, TempValue2)
		Cos256(TempValue1, TempValue2)

		Projectile.XVelocity = TempValue1
		Projectile.XVelocity *= -0x800

		Projectile.YVelocity = TempValue0
		Projectile.YVelocity *= -0x800
	else
		CallFunction(Player_Hit)
	end if
end function

// Called by block objects to see if Knuckles should climb them
// (Note that the block object needs to have called C_BOX3 before calling this function)
function Player_ClimbBlock
#platform: Use_Origins
	if Player.State != Player_State_Climb
		TempValue0 = false
		//ArrayPos0 = 32	// This goes with the commented-out code a few lines down, see below
		if Player.State == Player_State_GlideRightNoGrip
			TempValue0 = true
		else
			if Player.State == Player_State_GlideLeftNoGrip	// Original didn't check this, but might as well
				TempValue0 = true
			end if
			
			// Dev note: The following code doesn't seem like it would actually work, so I'm removing it for efficiency
			//while ArrayPos0 < 1056
			//	if Object[ArrayPos0].Type == TypeName[Blank Object] // huh? maybe this is supposed to be [NoGripArea], but you can't use stage object TypeNames in global objects...
			//		if Object[ArrayPos0].Value0 == true
			//			TempValue0 = true
			//		end if
			//	end if
			//	ArrayPos0++
			//loop
		end if

		if TempValue0 == false
			if Player.Direction == FACING_LEFT
				ObjectTileGrip(CSIDE_ENTITY, FACING_LEFT, false, ECEFFECT_BOXCOL3)
			else
				ObjectTileGrip(CSIDE_ENTITY, FACING_RIGHT, false, ECEFFECT_BOXCOL3)
			end if
			PlaySfx(34, false)	// SFX_G_GRAB
			Player.State = Player_State_Climb
			SetBit(Player.DelayLedgePull, 0, true)
			Player.Speed = 0
			Player.XVelocity = 0
			Player.YVelocity = 0
			Player.Timer = 0
			Player.Gravity = GRAVITY_AIR
			game.callbackParam0 = 2
			game.callbackParam1 = 0
			game.callbackParam2 = 0
			EngineCallback(NOTIFY_STATS_CHARA_ACTION2)
			if Warp.Destination > 0
				if Warp.Timer > 99
					if Warp.Timer < 204
						Warp.Destination = 0
					end if
				end if
				Warp.Timer = 0
			end if
		end if
	end if
#endplatform
end function

// These two functions are pretty similar, the former makes Knuckles cancel a climb into a glide drop, while the latter cancels a climb into normal air movement, instead

// See above, this one's p much only called by NoGripArea's
function Player_Unstick
#platform: Use_Origins
	CallFunction(Player_Action_GlideDrop)
	Player.PrevAnimation = ANI_GLIDING_DROP
	Player.Frame = 2
#endplatform
end function

// See above, this one's p much only called by Bumpers
function Player_CancelClimb
#platform: Use_Origins
	Player.Animation = ANI_WALKING
	Player.Timer = 0
	Player.State = Player_State_Air_NoDropDash
#endplatform
end function

// Called by No Grip Area objects to, well, force No Grip gliding
function Player_ForceNoGrip
#platform: Use_Origins
	if Player.State == Player_State_GlideLeft
		Player.State = Player_State_GlideLeftNoGrip
	end if
	if Player.State == Player_State_GlideRight
		Player.State = Player_State_GlideRightNoGrip
	end if
#endplatform

// Animation speed while running
function Player_HandleRunAnimSpeed
	if Player.Animation == ANI_PEELOUT
		Player.AnimationSpeed = 240	// top speed anims should always advance every frame
	else
		TempValue0 = Player.Speed
		if TempValue0 < 0
			FlipSign(TempValue0)
		end if
		
		if TempValue0 <= 0x60000
			Player.AnimationSpeed = 80
		else
			if TempValue0 >= 0x80000
				Player.AnimationSpeed = 240
			else
				TempValue0 >>= 12
				TempValue0 *= 5
				TempValue0 -= 400
				Player.AnimationSpeed = TempValue0
			end if
		end if
	end if
end function

// This makes you move
function Player_HandleGroundMovement
	if Player.ControlLock > 0
		Player.ControlLock--
		Sin256(TempValue0, Player.Angle)
		TempValue0   *= 0x2000
		TempValue0  >>= 8
		Player.Speed += TempValue0
	else
		if Player.Left == true
			TempValue0 = Player.TopSpeed
			FlipSign(TempValue0)
			if Player.Speed > TempValue0
				if Player.Speed > 0
					if Player.CollisionMode == CMODE_FLOOR
						// Skid
						if Player.Skidding == 0
							if Player.Speed > 0x40000
								if Player.Speed < 0x68000
									Player.Skidding = 16
								else
									Player.Skidding = 24
								end if
							end if
						end if
					end if

					if Player.Speed < Player.SkidSpeed
						Player.Speed 	= Player.SkidSpeed
						FlipSign(Player.Speed)
						Player.Skidding =  0
					else
						Player.Speed -= Player.SkidSpeed
					end if
				else
					Player.Speed   -= Player.Acceleration
					Player.Skidding = 0
				end if
			end if

			if Player.Speed <= 0
				Player.Direction = FACING_LEFT
			end if
		end if

		if Player.Right == true
			if Player.Speed < Player.TopSpeed
				if Player.Speed < 0
					if Player.CollisionMode == CMODE_FLOOR
						// Skid
						if Player.Skidding == 0
							if Player.Speed < -0x40000
								if Player.Speed > -0x68000
									Player.Skidding = 16
								else
									Player.Skidding = 24
								end if
							end if
						end if
					end if

					TempValue0 = Player.SkidSpeed
					FlipSign(TempValue0)
					if Player.Speed > TempValue0
						Player.Speed	= Player.SkidSpeed
						Player.Skidding = 0
					else
						Player.Speed += Player.SkidSpeed
					end if
				else
					Player.Speed   += Player.Acceleration
					Player.Skidding = 0
				end if
			end if

			if Player.Speed >= 0
				Player.Direction = FACING_RIGHT
			end if
		end if

		TempValue0  = Player.Left
		TempValue0 |= Player.Right
		if TempValue0 == false
			// tldr, reduce speed with deceleration, or based on angle
			// calculate deceleration on the fly to save on variable space
			TempValue0 = Player.Acceleration
			if Player.SpeedShoesTimer > 0
				TempValue0 >>= 1
			else
				if Player.SuperState == SUPERSTATE_SUPER
					TempValue0 >>= 2
				end if
			end if
			
			if Player.Speed > 0
				Player.Speed -= TempValue0	// was Player.Deceleration
				if Player.Speed < 0
					Player.Speed = 0
				end if
			else
				Player.Speed += TempValue0	// was Player.Deceleration
				if Player.Speed > 0
					Player.Speed = 0
				end if
			end if

			if Player.Speed > 0x2000
				Sin256(TempValue0, Player.Angle)
				TempValue0   *= 0x2000
				TempValue0  >>= 8
				Player.Speed += TempValue0
			end if

			if Player.Speed < -0x2000
				Sin256(TempValue0, Player.Angle)
				TempValue0   *= 0x2000
				TempValue0  >>= 8
				Player.Speed += TempValue0
			end if

			if Player.Angle > 192
				if Player.Angle < 228
					if Player.Speed > -0x10000
						if Player.Speed < 0x10000
							Player.ControlLock = 30 // Lock the player to move out of the angle
						end if
					end if
				end if
			end if

			if Player.Angle > 28
				if Player.Angle < 64
					if Player.Speed > -0x10000
						if Player.Speed < 0x10000
							Player.ControlLock = 30 // Lock the player to move out of the angle
						end if
					end if
				end if
			end if
		else
			Sin256(TempValue0, Player.Angle)
			TempValue0   *= 0x2000
			TempValue0  >>= 8
			Player.Speed += TempValue0

			if Player.Right == true
				if Player.Left == false
					if Player.Angle > 192
						if Player.Angle < 228
							if Player.Speed < 0x28000
								if Player.Speed > -0x20000
									Player.ControlLock = 30 // Lock the player to move out of the angle
								end if
							end if
						end if
					end if
				end if
			else
				if Player.Left == true
					if Player.Angle > 28
						if Player.Angle < 64
							if Player.Speed > -0x28000
								if Player.Speed < 0x20000
									Player.ControlLock = 30 // Lock the player to move out of the angle
								end if
							end if
						end if
					end if
				end if
			end if
		end if
	end if
end function

// Handle movement in the air
function Player_HandleAirFriction
	if Player.YVelocity > -0x40000
		if Player.YVelocity < 0
			TempValue0    = Player.Speed
			TempValue0  >>= 5
			Player.Speed -= TempValue0
		end if
	end if
	TempValue0 = Player.TopSpeed
	FlipSign(TempValue0)

	if Player.Speed > TempValue0
		if Player.Left == true
			Player.Speed -= Player.AirAcceleration
			Player.Direction = FACING_LEFT
		end if
	else
		if Player.Left == true
			Player.Direction = FACING_LEFT
		end if
	end if

	if Player.Speed < Player.TopSpeed
		if Player.Right == true
			Player.Speed += Player.AirAcceleration
			Player.Direction = FACING_RIGHT
		end if
	else
		if Player.Right == true
			Player.Direction = FACING_RIGHT
		end if
	end if

	if game.playMode == BOOT_PLAYMODE_CLASSIC
		if Player.Left == true
			TempValue0 = Player.TopSpeed
			FlipSign(TempValue0)
			if Player.Speed < TempValue0
				Player.Speed = TempValue0
			end if
		end if

		if Player.Right == true
			if Player.Speed > Player.TopSpeed
				Player.Speed = Player.TopSpeed
			end if
		end if
	end if
end function

// Handles desaceleration while rolling
function Player_HandleRollDeceleration

	if Player.Right == true
		if Player.Speed < 0
			Player.Speed += 0x2000	// was Player.RollingDeceleration
		end if
	end if

	if Player.Left == true
		if Player.Speed > 0
			Player.Speed -= 0x2000	// was Player.RollingDeceleration
		end if
	end if

	if Player.Speed > 0
		Player.Speed -= Player.RollingFriction	// was Player.AirDeceleration...?
		if Player.Speed < 0
			Player.Speed = 0
		end if

		if Player.Speed == 0
			if Player.Angle > 224
				Player.State = Player_State_Ground
			end if
			if Player.Angle < 32
				Player.State = Player_State_Ground
			end if
		end if

		Sin256(TempValue0, Player.Angle)
		if TempValue0 > 0
			Sin256(TempValue0, Player.Angle)
			TempValue0 *= 0x5000
		else
			Sin256(TempValue0, Player.Angle)
			TempValue0 *= 0x1E00
		end if

		TempValue0  >>= 8
		Player.Speed += TempValue0
	else
		Player.Speed += Player.RollingFriction	// was Player.AirDeceleration...?
		if Player.Speed > 0
			Player.Speed = 0
		end if

		if Player.Speed == 0
			if Player.Angle > 224
				Player.State = Player_State_Ground
			end if
			if Player.Angle < 32
				Player.State = Player_State_Ground
			end if
		end if

		Sin256(TempValue0, Player.Angle)
		if TempValue0 < 0
			Sin256(TempValue0, Player.Angle)
			TempValue0 *= 0x5000
		else
			Sin256(TempValue0, Player.Angle)
			TempValue0 *= 0x1E00
		end if

		TempValue0  >>= 8
		Player.Speed += TempValue0
	end if

	if Player.Speed > 0x180000
		Player.Speed = 0x180000
	end if

	if Player.Speed < -0x180000
		Player.Speed = -0x180000
	end if
end function

// Movement when you fall from a cliff
function Player_HandleAirMovement
	Player.TrackScroll = true

	Player.YVelocity += Player.GravityStrength
	if Player.YVelocity < Player.JumpCap
		if Player.JumpHold == false
			if Player.Timer > 0
				Player.YVelocity  = Player.JumpCap
				TempValue0 		  = Player.Speed
				TempValue0		>>= 5
				Player.Speed	 -= TempValue0
			end if
		end if
	end if
	Player.XVelocity = Player.Speed

	if Player.Rotation < 256
		if Player.Rotation > 0
			Player.Rotation -= 4
		else
			Player.Rotation = 0
		end if
	else
		if Player.Rotation < 512
			Player.Rotation += 4
		else
			Player.Rotation = 0
		end if
	end if

	Player.CollisionMode = CMODE_FLOOR
	if Player.Animation == ANI_JUMPING
		Player.AnimationSpeed = Player.RollAnimationSpeed
	end if
end function

		// Added in this mod, used to correct some issues with jumping being from the player's center point.
		if Options.AttractMode == false
			if Player.Animation != ANI_JUMPING
				Player.iYPos -= Player.JumpOffset
				TempValue0 += Player.JumpOffset
			end if
		end if
#endplatform

		Object.XPos = TempValue6
		Object.YPos = TempValue7
	end if

	if CheckResult == false
		Player.ControlLock	= 0
		Player.Gravity		= GRAVITY_AIR
		Player.AbilityTimer = 8
		// Calculation adjustments were made here to make slope-jumps just like 1993.
		if Options.AttractMode == false // This is needed to fix demos.
			Sin256(TempValue0, Player.Angle)
			TempValue0 *= Player.JumpStrength
			TempValue0 >>= 8
			Player.XVelocity += TempValue0
		
			Cos256(TempValue0, Player.Angle)
			TempValue0 *= Player.JumpStrength
			TempValue0 >>= 8
			TempValue0 += Player.GravityStrength // this is needed to correct the jumping to what it should be
			Player.YVelocity -= TempValue0
		else
			Sin256(Player.XVelocity, Player.Angle)
			Player.XVelocity  *= Player.JumpStrength
			Cos256(TempValue0, Player.Angle)
			TempValue0		  *= Player.Speed
			Player.XVelocity  += TempValue0
			Player.XVelocity >>= 8

			Sin256(Player.YVelocity, Player.Angle)
			Player.YVelocity  *= Player.Speed
			Cos256(TempValue0, Player.Angle)
			TempValue0		  *= Player.JumpStrength
			Player.YVelocity  -= TempValue0
			Player.YVelocity >>= 8
		end if

		Player.Speed		 = Player.XVelocity
		Player.TrackScroll   = true
		Player.Animation	 = ANI_JUMPING
		Player.Angle		 = 0
		Player.CollisionMode = CMODE_FLOOR
		Player.Timer		 = 1
		CallFunction(Player_HandleRollAnimSpeed)

		Player.State = Player_State_Air
#platform: Use_Origins
		Player.JumpAbilityState = 1 // Allow for air abilities! This was missing from CD despite ST adding the S1 ledge roll off detection code.
#endplatform

		PlaySfx(0, false)	// SFX_G_JUMP
		if Options.AttractMode == false
			Player.OffsetJump = true
		end if
	end if

#platform: Use_Origins
	if game.playMode == BOOT_PLAYMODE_CLASSIC
		Player.DropDashCharge = -1
	else
		if Options.AttractMode == true
			Player.DropDashCharge = -1
		else
			Player.DropDashCharge =  0
		end if
	end if
#endplatform
end function

// Epic real authentic drop dash!!!!
function Player_Action_Dropdash
	TempValue3 = Player.XVelocity
	
	Player.Skidding = 0
	Player.Timer = 0
	Player.State = Player_State_Roll
	Player.Animation = ANI_JUMPING
	
	Player.DropDashCharge = -1

	if Options.OriginalControls == true
		Screen.CameraStyle = CAMERASTYLE_EXTENDED
	else
		Player.ScrollDelay = 8
		Screen.CameraStyle = CAMERASTYLE_HLOCKED
	end if
	
	StopSfx(28)	// SFX_G_DROPDASH
	PlaySfx(7, false)	// SFX_G_RELEASE
	
	// speed math starts here
	
	if Player.SuperState == SUPERSTATE_SUPER
		TempValue0 = 0xC0000
		TempValue1 = 0xD0000
	else
		TempValue0 = 0x80000
		TempValue1 = 0xC0000
	end if
	
	TempValue2 = Player.Speed
	
	if Player.Direction == FACING_LEFT
		FlipSign(TempValue2)
		FlipSign(TempValue3)
	end if
	
	if TempValue3 >= 0
		TempValue2 >>= 2
		TempValue2 += TempValue0
		if TempValue2 > TempValue1
			TempValue2 = TempValue1
		end if
	else
		if Player.Angle == 0
			TempValue2 = TempValue0
		else
			TempValue2 >>= 1
			TempValue2 += TempValue0
		end if
	end if
	
	if Player.Direction == FACING_LEFT
		FlipSign(TempValue2)
	end if
	
	Player.Speed = TempValue2
	
	// speed math ends here

	// Create dust puff object
	CreateTempObject(TypeName[Dust Puff], 0, Player.XPos, Player.YPos)
	
	if Mini_PlayerFlag == true
		Object[TempObjectPos].YPos += 0xB0000
	else
		Object[TempObjectPos].YPos += 0xE0000
	end if
	
	Object[TempObjectPos].Frame = 11
	Object[TempObjectPos].DrawOrder  = Player.DrawOrder
	Object[TempObjectPos].Direction = Player.Direction
	
	CallFunction(Player_HandleOnGround)
end function

function Player_State_Air
	CallFunction(Player_HandleAirFriction)
#platform: Use_Origins
	if Stage.PlayerListPos == PLAYER_TAILS
		CallFunction(Player_HandleDropDash)
	end if
#endplatform
	if Player.Gravity == GRAVITY_AIR	// Check that you're truly in the air
		CallFunction(Player_HandleAirMovement)

		// More warp stuff
		if Warp.Destination > WARPDEST_NONE
			TempValue0 = Player.YVelocity
			if TempValue0 < 0
				FlipSign(TempValue0)
			end if

			if TempValue0 < 0x60000
				TempValue0 = Player.XVelocity
				if TempValue0 < 0
					FlipSign(TempValue0)
				end if

				if TempValue0 < 0x60000
					TempValue0  = Player.XVelocity
					TempValue0 -= Warp.SpeedCompare
					if TempValue0 < 0
						FlipSign(TempValue0)
					end if

					if TempValue0 > 0x40000
						if Warp.Timer > 99
							if Warp.Timer < 204
								Warp.Destination = WARPDEST_NONE
							end if
						end if
						Warp.Timer = 0
					end if
				end if
			end if
			Warp.SpeedCompare = Player.XVelocity
		end if

		// changes the flailing with the walking, not sure when this is used
		if Player.YVelocity > 0x20000
			if Player.Animation == ANI_FLAILINGLEFT
				Player.Animation = ANI_WALKING
			end if
			if Player.Animation == ANI_FLAILINGRIGHT
				Player.Animation = ANI_WALKING
			end if
		end if

		// Post-Hitting a spring
		if Player.Animation == ANI_BOUNCING
			if Player.YVelocity >= 0
				if Player.AnimationReserve == ANI_STOPPED
					Player.AnimationReserve = ANI_WALKING

#platform: Use_Origins

		end if

			else

			end if

	else
		// go back to the ground silly
#platform: Use_Origins
		if Player.DropDashCharge >= 21
			if Stage.PlayerListPos == PLAYER_AMY

			else
				CallFunction(Player_Action_Dropdash)
			end if

			end if
		else

			CallFunction(Player_HandleOnGround)
			Player.Skidding = 0
		end if
#endplatform

#platform: Use_Standalone

// S2 Spin Dash code
function Player_State_Spindash_S2
	// Keep the camera lock, cancel the spin dash
	if Player.Gravity == GRAVITY_AIR // this check was altered to allow for spin-dash ledge dropping.
		Player.Down = true
		Player.JumpPress = false
		CallFunction(Player_HandleAirMovement)
		CallFunction(Player_HandleAirFriction)
	end if

	if Player.Gravity == GRAVITY_GROUND // if you're on the ground, stop your speed while in the spindash state.
		Player.Speed = 0				// this should work in all cases, and not cause bugs i hope.
	end if

	// Mashing
	if Player.JumpPress == true
		if Player.AbilityTimer < 0x90000
			Player.AbilityTimer += 0x20000
		else
			Player.AbilityTimer = 0x80000
		end if
		
		Player.Frame = 0
		PlaySfx(6, false)	// SFX_G_CHARGE
	else
		TempValue0 = Player.AbilityTimer
		TempValue0 >>= 5
		Player.AbilityTimer -= TempValue0
	end if
	
	if SegaID.LoggedIn == 0
		SegaID.LoggedIn = -1 // Used to pause time travel (Knuckles be like:)
	end if
	
	// Spin Dash release
	if Player.Down == false
		Player.Timer	 = 0
		Player.State	 = Player_State_Roll
		Player.Animation = ANI_JUMPING
		Player.iYPos	-= Player.JumpOffset

		Player.ScrollDelay = 15
		Screen.CameraStyle = CAMERASTYLE_HLOCKED

		TempValue0	 = Player.AbilityTimer
		TempValue0 >>= 17
		TempValue0 <<= 16
		
		if Player.SuperState == SUPERSTATE_SUPER
			TempValue0 += 0xB0000
		else
			TempValue0 += 0x80000
		end if

		if Player.Direction == FACING_RIGHT
			Player.Speed = TempValue0
		else
			Player.Speed = TempValue0
			FlipSign(Player.Speed)
		end if

		PlaySfx(7, false)	// SFX_G_RELEASE
		
		SegaID.LoggedIn = 0 // Allow time travel!
		
		CallFunction(Player_HandleOnGround)

// #platform: Use_Origins
// 		Player.ForceGrounded = 6
// #endplatform

#platform: Use_Haptics
		HapticEffect(42, 0, 0, 0)
#endplatform

	end if
end function

// Tails flight
function Player_State_Fly
	CallFunction(Player_HandleAirFriction)
	if Player.Gravity == GRAVITY_AIR
		Player.XVelocity = Player.Speed
		// Check if you have enough speed while flying to keep the warp
		if Warp.Destination > WARPDEST_NONE
			TempValue0 = Player.XVelocity
			if TempValue0 < 0
				FlipSign(TempValue0)
			end if

			TempValue1 = Player.YVelocity
			if TempValue1 < 0
				FlipSign(TempValue1)
			end if

			TempValue0 += TempValue1
			if TempValue0 < 0x40000
				if Warp.Timer > 99
					if Warp.Timer < 204
						Warp.Destination = WARPDEST_NONE
					end if
				end if
				Warp.Timer = 0
			end if
		end if

		if Player.YVelocity < -0x10000
			Player.FlightVelocity = 0x800
		else
			if Player.YVelocity < 1
				if Player.AbilityTimer < 60
					Player.AbilityTimer++
				else
					Player.FlightVelocity = 0x800
				end if
			end if
		end if

		Player.YVelocity += Player.FlightVelocity

		if Player.YPos < 0x100000 // new to Origins Ultrafix, makes tails flight more accurate to the genesis games.
			if Player.YVelocity < 0
				Player.YVelocity = 0
			end if
		end if

		if Player.Timer < 480
			if Player.GravityStrength == 0x3800
				Player.Animation = ANI_FLYING
			else
				Player.Animation = ANI_SWIMMING
			end if

			Player.Timer++
			if Player.Timer == 480
				if Player.GravityStrength == 0x3800
					Player.Animation = ANI_FLYINGTIRED
					StopSFX(24)	// SFX_G_FLYING
					PlaySfx(25, true)	// SFX_G_TIRED
				else
					Player.Animation = ANI_SWIMMINGTIRED
				end if
			else
				if Player.JumpPress == true
					Player.FlightVelocity	= -0x2000
					Player.AbilityTimer		= 0
				end if
			end if
		else
			if Player.GravityStrength == 0x3800
				Player.Animation = ANI_FLYINGTIRED
			else
				Player.Animation = ANI_SWIMMINGTIRED
			end if
		end if
	else
		Player.State = Player_State_Ground
		CallFunction(Player_HandleOnGround)
	end if

	// If the level has a roof barrier, you're not getting too far
	if Player.RoofBarrier == true
		TempValue0   = Player.YPos
		TempValue0 >>= 16
		if TempValue0 < Player.CollisionBottom
			Player.YPos   = Player.CollisionBottom
			Player.YPos <<= 16
		end if
	end if
end function

// Player Damage
function Player_State_GotHit
#platform: Use_Origins
	TempValue1 = false
	if game.playMode == BOOT_PLAYMODE_MISSION
		if game.missionFunctionNo == RULE_MERCY_BADNIKS
			if game.missionValue == 1
				game.missionValue = 0
				TempValue1 = true
			end if
		end if
	end if
#endplatform

	ArrayPos0  = Player.EntityNo
	ArrayPos0 += 2
	// if you had a shield, remove it
	if Object[ArrayPos0].PropertyValue > 0
		TempValue0 = DAMAGE_SHIELDED
		ResetObjectEntity(ArrayPos0, TypeName[Blank Object], 0, 0, 0)
		PlaySfx(5, false)	// SFX_G_HURT
	else
		if Player.Rings == 0
			PlaySfx(5, false)	// SFX_G_HURT
			TempValue0 = DAMAGE_DEATH
		else
#platform: Use_Origins
			if TempValue1 == true
				PlaySfx(5, false)	// SFX_G_HURT
			else
				PlaySfx(4, false)	// SFX_G_LOSERINGS
			end if
#endplatform

#platform: Use_Standalone
			PlaySfx(4, false)	// SFX_G_LOSERINGS
#endplatform
			TempValue0 = DAMAGE_HURT
		end if
	end if

#platform: Use_Origins
	if TempValue1 != false
		TempValue0 = DAMAGE_DEATH
	end if
#endplatform

	Player.Visible = true

	switch TempValue0
	case DAMAGE_SHIELDED
#platform: Use_Haptics
		HapticEffect(16, 0, 0, 0)
#endplatform

		Player.State			= Player_State_Hurt
		Player.Animation		= ANI_HURT
		Player.YVelocity		= -0x40000
		Player.Gravity			= GRAVITY_AIR
		Player.TrackScroll		= true
		Player.InvincibleTimer	= 8000

		if Player.GravityStrength == 0x1000
			Player.Speed 	 >>= 1
			Player.YVelocity >>= 1
		end if
		break

	case DAMAGE_HURT
#platform: Use_Haptics
		HapticEffect(16, 0, 0, 0)
#endplatform

		if Player.CollisionPlane == COLLISION_PLANE_A
			TempValue4 = 3
		else
			TempValue4 = 1
		end if

		Player.State			= Player_State_Hurt
		Player.Animation		= ANI_HURT
		Player.YVelocity		= -0x40000
		Player.Gravity			= GRAVITY_AIR
		Player.TrackScroll		= true
		Player.InvincibleTimer	= 8000

		if Player.GravityStrength == 0x1000
			Player.Speed	 >>= 1
			Player.YVelocity >>= 1
		end if
		// Lose ring code
		TempValue0 = Player.Rings
		if TempValue0 > 16
			TempValue1  = TempValue0
			TempValue1 -= 16
			TempValue0  = 16
		else
			TempValue1 = 0
		end if

		if TempValue1 > 16
			TempValue1 = 16
		end if

		TempValue3   = TempValue1
		TempValue3 >>= 1
		TempValue3 <<= 5

		TempValue2  = 384
		TempValue2 -= TempValue3

		TempValue3 >>= 4
		if TempValue3 == TempValue1
			TempValue2 += 16
		else
			TempValue2 -= 16
		end if

		TempValue3 = 0
		while TempValue3 < TempValue1
			CreateTempObject(TypeName[Lose Ring], Player.CollisionPlane, Player.XPos, Player.YPos)
			Cos(Object[TempObjectPos].XVelocity, TempValue2)
			Sin(Object[TempObjectPos].YVelocity, TempValue2)
			Object[TempObjectPos].XVelocity		<<= 8
			Object[TempObjectPos].YVelocity		<<= 8
			Object[TempObjectPos].DrawOrder		  = TempValue4
			Object[TempObjectPos].AnimationSpeed  = 256
			Object[TempObjectPos].InkEffect = 2 // INK_ALPHA
			Object[TempObjectPos].Alpha = 255
			TempValue3++
			TempValue2 += 32
		loop
		TempValue3   = TempValue0
		TempValue3 >>= 1
		TempValue3 <<= 5

		TempValue2  = 384
		TempValue2 -= TempValue3

		TempValue3 >>= 4
		if TempValue3 == TempValue0
			TempValue2 += 16
		else
			TempValue2 -= 16
		end if
		TempValue3 = 0

		while TempValue3 < TempValue0
			CreateTempObject(TypeName[Lose Ring], Player.CollisionPlane, Player.XPos, Player.YPos)
			Cos(Object[TempObjectPos].XVelocity, TempValue2)
			Sin(Object[TempObjectPos].YVelocity, TempValue2)
			Object[TempObjectPos].XVelocity		<<= 9
			Object[TempObjectPos].YVelocity		<<= 9
			Object[TempObjectPos].DrawOrder		  = TempValue4
			Object[TempObjectPos].AnimationSpeed  = 256
			Object[TempObjectPos].InkEffect = 2 // INK_ALPHA
			Object[TempObjectPos].Alpha = 255
			TempValue3++
			TempValue2 += 32
		loop
		Player.Rings = 0
		Ring.ExtraLife = 100
		break

	case DAMAGE_DEATH
#platform: Use_Haptics
		HapticEffect(28, 0, 0, 0)
#endplatform

		Object.DrawOrder = 5
		Player.Speed	 = 0
		Player.YVelocity = -0x70000
		Player.XVelocity = 0
		Player.State	 = Player_State_Death
		Player.Animation = ANI_DYING

		Player.TileCollisions	 = false
		Player.ObjectInteraction = false
		if Player.EntityNo == 0	// Check if it's was player 1 who died......what?
			Screen.CameraEnabled = false
		end if
		break
	end switch
	// lose your warp-run if you receive damage
	if Warp.Destination > WARPDEST_NONE
		if Warp.Timer > 99
			if Warp.Timer < 204
				Warp.Destination = WARPDEST_NONE
			end if
		end if
		Warp.Timer = 0
	end if
end function

// YOU DIED.
function Player_State_Death
	//if player.entityPos == SLOT_PLAYER1
		if Player.SuperState == SUPERSTATE_SUPER
			Player.SuperState = SUPERSTATE_UNTRANSFORM
		end if
	//end if

	Player.ControlMode = CONTROLMODE_NONE
	Player.YVelocity  += 0x3800
	// Don't override the bored animation
	if Player.Animation != ANI_BORED
		Player.Animation = ANI_DYING
	end if

	if Player.BlinkTimer != 0
        Player.BlinkTimer = 0
        Player.Visible = true
    end if

	if Player.YVelocity > 0x100000
#platform: Use_Origins
		EngineCallback(NOTIFY_DEATH_EVENT)
#endplatform

		if Player.Lives > 0
			if Player.Animation == ANI_BORED	// Outta Here makes an automatic game over, no lifes will save you
				Player.Lives = 0
			else

#platform: Use_Origins
				// Check that we are actually in a mode where we use lives
				if game.coinMode == false
					if game.playMode != BOOT_PLAYMODE_MISSION
						CheckEqual(game.playMode, BOOT_PLAYMODE_BOSSRUSH)
						TempValue0 = CheckResult
						CheckEqual(game.oneStageFlag, false)
						TempValue0 |= CheckResult
						if TempValue0 != 0
							Player.Lives--
						end if
					end if
				end if
#endplatform

#platform: Use_Standalone
				// In Standalone, we don't have any extra things to worry about
				Player.Lives--
#endplatform
			end if
		end if

		Stage.TimeEnabled = false

#platform: Use_Origins
		CheckEqual(game.playMode, BOOT_PLAYMODE_BOSSRUSH)
		TempValue0   = CheckResult
		CheckNotEqual(Player.Lives, 0)
		TempValue0  |= CheckResult
		// Check if the player is replaying the stage from My Data & Rankings
		CheckEqual(game.oneStageFlag, false)
		CheckResult |= TempValue0
#endplatform

#platform: Use_Standalone
		// Slight workaround to make this cleaner - the check below this comes from Origins, but we need it to always be true on standalone

		CheckResult = true
#endplatform

		// Again, this check is new to Origins and doesn't exist in Standalone
		if CheckResult != false
			Object.Type = TypeName[Death Event]
			DeathEvent.DrawOrder  = 7
			DeathEvent.Text1XPos  = Screen.CenterX
			DeathEvent.Text1XPos -= 232
			DeathEvent.Text2XPos  = Screen.CenterX
			DeathEvent.Text2XPos += 232
		end if

		if Options.GameMode == MODE_TIMEATTACK
			DeathEvent.Timer = 0
			DeathEvent.State = DEATHEVENT_TIMEATTACK
		else
			if Player.Lives == 0
				DeathEvent.Timer = -2880
				DeathEvent.State = DEATHEVENT_GAMEOVER
				// *Technically it goes to the traditional game over in the same state
				PlayMusic(5)
				Stage.PauseEnabled = false

#platform: Use_Origins
				if game.oneStageFlag != false
					game.callbackResult = -1				// Reset any callback result
					game.callbackParam0 =  0				// 0 - Normal Retry, anything else - Show current and best time
					game.callbackParam1 =  Stage.ListPos	// Get current stage
					game.callbackParam2 =  0				// Unknown
					EngineCallback(NOTIFY_STAGE_RETRY)
				end if
#endplatform

			else
				DeathEvent.Timer = 0
				DeathEvent.State = DEATHEVENT_FADEOUT

#platform: Use_Origins
				// Lots of checks here, before we do anything else

				if game.coinMode == false
					if game.playMode != BOOT_PLAYMODE_BOSSRUSH
						if Object[23].Value0 == true
							// Check if the player is replaying the stage from My Data & Rankings
							if game.oneStageFlag == false
								DeathEvent.Timer = -2880
								DeathEvent.State = DEATHEVENT_TIMEOVER

								PlayMusic(5)
								Stage.PauseEnabled = false
							end if
						end if
					end if
				end if
#endplatform

#platform: Use_Standalone
				if Stage.Minutes == 9
					if Stage.Seconds == 59
						DeathEvent.Timer = -2880
						DeathEvent.State = DEATHEVENT_TIMEOVER

						PlayMusic(5)
						Stage.PauseEnabled = false
					end if
				end if
#endplatform
			end if
		end if
	end if
end function

		if Player.SuperState == SUPERSTATE_SUPER
			Player.SuperState = SUPERSTATE_UNTRANSFORM
		end if
	//end if

	Player.ControlMode = CONTROLMODE_NONE

	Player.Animation   = ANI_DROWNING

	if Player.YVelocity > 0x80000
#platform: Use_Origins
		EngineCallback(NOTIFY_DEATH_EVENT)

		// We only subtract lives sometimes in Origins
		if game.coinMode == false
			if game.playMode != BOOT_PLAYMODE_MISSION

					end if
				end if
			end if
		end if
#endplatform

		end if
#endplatform

		Stage.TimeEnabled = false

// Wacky Workbench hanging bar gimmick
function Player_State_HangBar
	if Player.Left == true
		Player.Direction	  = FACING_LEFT
		Player.Speed		  = -0x20000
		Player.AnimationSpeed = 30
	else
		if Player.Right == true
			Player.Direction	  = FACING_RIGHT
			Player.Speed		  = 0x20000
			Player.AnimationSpeed = 30
		else
			Player.Speed		  = 0
			Player.AnimationSpeed = 0
		end if
	end if

	TempValue1   = Player.XPos
	TempValue1 >>= 16

	TempValue2   = Player.YPos
	TempValue2 >>= 16
	TempValue2  += Player.CollisionTop
	// Check that you are in the hanging bar
	Get16x16TileInfo(TempValue0, TempValue1, TempValue2, 8) // TILEINFO_ANGLEB on the last number here
	if TempValue0 != 3	// seems like you're not

#platform: Use_Origins
		Player.State = Player_State_Air_NoDropDash
#endplatform

#platform: Use_Standalone
		Player.State = Player_State_Air
#endplatform

		Player.Speed		  = 0
		Player.AnimationSpeed = 0
		Player.YVelocity	  = 0
	end if

	// Manual Drop
	if Player.JumpPress == true

		Player.State		  = Player_State_Air
		Player.YVelocity	  = 0
		Player.Speed		  = 0
		Player.AnimationSpeed = 0
		Player.YPos += 0x40000
	end if
	Player.XVelocity = Player.Speed
	// Cancel warp
	if Warp.Destination > WARPDEST_NONE
		if Warp.Timer > 99
			if Warp.Timer < 204
				Warp.Destination = WARPDEST_NONE
			end if
		end if

		Warp.Timer = 0
	end if
end function

// Palmtree Panic Ramp
function Player_State_Ramp3D
	CallFunction(Player_HandleGroundMovement)
	if Player.Gravity == GRAVITY_AIR

#platform: Use_Origins
		Player.State = Player_State_Air_NoDropDash
#endplatform

#platform: Use_Standalone
		Player.State = Player_State_Air
#endplatform

		Player.Timer = 0
		CallFunction(Player_HandleAirMovement)
		// Set the equivalent animation for looking to the left
		if Player.Direction == FACING_LEFT

			if Player.Animation == ANI_RAMP_RUNNING3
				Player.Animation = ANI_RAMP_RUNNING5
			end if

			if Player.Animation == ANI_RAMP_RUNNING2
				Player.Animation = ANI_RAMP_RUNNING6
			end if

		end if
	else
#platform: Use_Origins
		if Player.Animation == ANI_GLIDING
			Player.XVelocity = 0
			Player.YVelocity = 0
			Player.Speed = 0
			Player.Animation = ANI_GLIDING_STOP
		end if
#endplatform
		CallFunction(Player_HandleOnGround)
		TempValue0 = Player.Speed
		if TempValue0 < 0
			FlipSign(TempValue0)
		end if
		// Warp-Consuming
		if TempValue0 < 0x5F5C2
			// Warp Star erase check
			if Warp.Destination > WARPDEST_NONE
				if Warp.Timer > 99
					if Warp.Timer < 204
						Warp.Destination = WARPDEST_NONE
					end if
				end if
				Warp.Timer = 0
			end if
		else
			// Spawn Warp Stars
			if Warp.Destination > WARPDEST_NONE
				if Warp.Timer == 0
					Warp.Timer			= 1
					ResetObjectEntity(3, TypeName[Warp Star], 0, Player.XPos, Player.YPos)
					WarpStar.Timer		= 7
					WarpStar.DrawOrder	= 4
				end if
			end if
		end if
		if Player.Speed == 0
			// Waiting timer while in the ramp
			switch Stage.PlayerListPos
			case PLAYER_SONIC_A
				CheckNotEqual(Player.SuperState, SUPERSTATE_SUPER)
				TempValue0 = CheckResult
				TempValue0 |= Mini_PlayerFlag
				
				// Original behaviour - Outta Here (Sonic, Classic Mode only)
				if Player.Timer < 240
					if TempValue0 == false
						Player.Animation = ANI_WAITING
					else
						Player.Animation = ANI_STOPPED
					end if
					Player.Timer++
				else	// Waiting animation
					if Player.SuperState == SUPERSTATE_NONE
						Player.Animation = ANI_WAITING
						if game.playMode == BOOT_PLAYMODE_CLASSIC
							Player.Timer++
							if Player.Timer == 10620 // 177 seconds (2 minutes, 57 seconds)
								Player.Timer = 0

								// This SFX is muted in Origins
								PlaySfx(26, false)	// SFX_G_OUTTAHERE

								Player.State	 = Player_State_OuttaHere
								Player.Animation = ANI_BORED
							end if
						end if
					else
						if Player.SuperState == SUPERSTATE_SUPER
							Player.Animation = ANI_WAITING
						else
							if Mini_PlayerFlag == false
								Player.Animation = ANI_STOPPED
							end if
						end if
					end if
				end if
				break
				
			case PLAYER_TAILS_A
				if Player.Timer < 240
					Player.Animation = ANI_STOPPED
					Player.Timer++
				else	// Waiting animation
					Player.Animation = ANI_WAITING
				end if
				break
			
			case PLAYER_KNUCKLES_A
				// Knuckles sparring with the air
				if Player.Timer < 240
					Player.Animation = ANI_STOPPED
					Player.Timer++
				else
					if Player.Timer < 570
						Player.Animation = ANI_WAITING
						Player.Timer++
					else
						if Mini_PlayerFlag == false
							Player.Animation = ANI_BORED
							Player.Timer++
							if Player.Timer == 842
								Player.Timer = 0
								Player.Animation = ANI_STOPPED
							end if
						else
							Player.Timer = 0
							Player.Animation = ANI_STOPPED
						end if
					end if
				end if
				break
				
			case PLAYER_AMY_A
				// Amy daydreaming
				if Player.Timer < 240
					Player.Animation = ANI_STOPPED
					Player.Timer++
				else
					if Player.Timer < 1107
						Player.Animation = ANI_WAITING
						Player.Timer++
					else
						if Mini_PlayerFlag == false
							Player.Animation = ANI_BORED
							Player.Timer++
							if Player.Timer >= 1152
								Player.Timer = 1107
							end if
						end if
					end if
				end if
				break
			end switch
		else
			// Set animation based on angle
			if Player.Angle == 0
				Player.Animation = ANI_RAMP_RUNNING1
			end if

			if Player.Angle > 200
				if Player.Direction == FACING_RIGHT
					Player.Animation = ANI_RAMP_RUNNING3
				else
					Player.Animation = ANI_RAMP_RUNNING5
				end if
			end if

			if Player.Angle > 216
				if Player.Direction == FACING_RIGHT
					Player.Animation = ANI_RAMP_RUNNING2
				else
					Player.Animation = ANI_RAMP_RUNNING6
				end if
			end if

			if Player.Angle > 232
				Player.Animation = ANI_RAMP_RUNNING1
			end if

			if Player.Angle == 192
				Player.Animation = ANI_RAMP_RUNNING4
			end if
		end if

		if Player.JumpPress == true
			CallFunction(Player_Action_Jump)
		else
			if Player.Up == true
				if Player.Speed == 0
					Player.State	 = Player_State_LookUp
					Player.Timer	 = 0
				end if
			end if
			if Player.Down == true
				if Player.Speed == 0
					Player.State	 = Player_State_Crouch
					Player.Timer	 = 0
				else
					if Player.Speed >  0x199A
						Player.ControlLock	= 0
						Player.State		= Player_State_Roll
						Player.Animation	= ANI_JUMPING
						if game.playMode != BOOT_PLAYMODE_CLASSIC
							PlaySfx(45, 0)
						end if
					end if

					if Player.Speed < -0x199A
						Player.ControlLock	= 0
						Player.State		= Player_State_Roll
						Player.Animation	= ANI_JUMPING
						if game.playMode != BOOT_PLAYMODE_CLASSIC
							PlaySfx(45, 0)
						end if
					end if
				end if
			end if
		end if
	end if
	
	if Player.Animation == ANI_STOPPED
		Player.AnimationSpeed = 0
	else
		if Player.Animation != ANI_WAITING
			if Player.Animation != ANI_BORED
				TempValue0 = false

				CheckEqual(Player.Animation, ANI_RAMP_RUNNING1)
				TempValue0 = CheckResult
				CheckEqual(Player.Animation, ANI_RAMP_RUNNING2)
				TempValue0 |= CheckResult
				CheckEqual(Player.Animation, ANI_RAMP_RUNNING3)
				TempValue0 |= CheckResult
				CheckEqual(Player.Animation, ANI_RAMP_RUNNING4)
				TempValue0 |= CheckResult
				CheckEqual(Player.Animation, ANI_RAMP_RUNNING5)
				TempValue0 |= CheckResult
				CheckEqual(Player.Animation, ANI_RAMP_RUNNING6)
				TempValue0 |= CheckResult
				
				if TempValue0 == true
					TempValue0 = Player.Speed
					if TempValue0 < 0
						FlipSign(TempValue0)
					end if
					
					CheckNotEqual(Player.SuperState, SUPERSTATE_SUPER)
					TempValue1 = CheckResult
					CheckLower(TempValue0, 0x5F5C2)
					TempValue1 &= CheckResult
						
					if TempValue1 == true
						CallFunction(Player_HandleWalkAnimSpeed)
					else
						CallFunction(Player_HandleRunAnimSpeed)
					end if
				else
					CallFunction(Player_HandleWalkAnimSpeed)
				end if
			end if
		end if
	end if
end function

function Player_HandleDropDash
#platform: Use_Origins
	if Player.DropDashCharge >= 0
		if Player.DropDashCharge == 0
			if Player.JumpPress == true
				if Stage.PlayerListPos == PLAYER_SONIC
					CheckEqual(Object[+2].Type, TypeName[Blue Shield])
					TempValue0 = CheckResult
					CheckGreater(Object[+2].ShieldType, SHIELD_INSTA)
					TempValue0 &= CheckResult
					if TempValue0 == true
						Player.DropDashCharge = -1
					else
						Player.DropDashCharge =  1
					end if
				else
					if Stage.PlayerListPos == PLAYER_AMY
						Player.DropDashCharge = 1
					else
						Player.DropDashCharge = -1
					end if
				end if
			end if
		else
			if Player.JumpHold == false
				if Stage.PlayerListPos == PLAYER_TAILS
					Player.DropDashCharge = -1
				else
					if Player.DropDashCharge >= 21
						Player.DropDashCharge = -1
						if Stage.PlayerListPos == PLAYER_SONIC_A
							Player.Animation = ANI_JUMPING
							Player.RollAnimationSpeed = 240
						else
							if Stage.PlayerListPos == PLAYER_AMY_A
								StopSfx(28)	// SFX_G_DROPDASH	// this happens in S3K so :shrug:
							end if
						end if
					end if
				end if
			else
				Player.DropDashCharge++
				if Player.DropDashCharge == 21
					PlaySfx(28, false)	// SFX_G_DROPDASH
					
					if Stage.PlayerListPos == PLAYER_AMY_A
						if Player.Animation == ANI_JUMPING	// should normally always already be ANI_HAMMER_JUMP, but this can happen under rare circumstances
							
							// and f u I'm doing it the proper way here
							Player.Animation = ANI_HAMMER_JUMP
							Player.PrevAnimation = ANI_HAMMER_JUMP
							
							// adjust frame to match hammer jump frame
							if Player.Frame == 4
								Player.Frame = 7
							else
								Player.Frame <<= 1
							end if
							
							PlaySfx(29, false)	// SFX_G_HAMMERJUMP
							
						end if
					else
						Player.Animation = ANI_HAMMER_JUMP	// this is actually the drop dash animation because I don't feel like adding a new alias
					end if
					
				end if
			end if
		end if
	end if
#endplatform
end function

function Player_Action_DblJumpKnux
#platform: Use_Origins
	// Handle the super activation
	GetBit(CheckResult, secrets.gameplay, SECRET_GP_SUPERFORMS)
	if CheckResult == true
		if KeyPress.ButtonY != false
			// Following checks are to see if Knuckles should transform instead of doing arial abilities

			// Each bit represents an emerald's collected state - 0x7F is 7 1's, AKA 7 emeralds
			CheckEqual(SpecialStage.TimeStones, 0x7F)
			TempValue0 = CheckResult

			// Player needs to have at least 50 rings
			CheckGreater(Player.Rings, 49)
			TempValue0 &= CheckResult

			// Player can't already be Super, of course
			CheckNotEqual(Player.SuperState, SUPERSTATE_SUPER)
			TempValue0 &= CheckResult

			// Don't let the player transform if the level is over, either
			CheckNotEqual(Object[30].Type, TypeName[ActFinish])
			TempValue0 &= CheckResult
			
			if TempValue0 == true
				// All checks passed, transform!
				CallFunction(Player_TryTransform)
			end if
		end if
	end if
	
	if Player.JumpPress == true
		CheckNotEqual(Player.State, Player_State_Transform) // listen... this is the only way i could get this shit to work since the transform key is always a jump press. :sobs:
		if CheckResult == true

			game.callbackParam0 = 1
			game.callbackParam1 = 0
			game.callbackParam2 = 0
			EngineCallback(NOTIFY_STATS_CHARA_ACTION2)
			Player.Speed = 0x40000
			Player.YVelocity += 0x20000 // Added here in this mod, restores glide bouncing.
			if Player.YVelocity < 0
				Player.YVelocity = 0
			end if
			if Player.Direction == FACING_RIGHT
				Player.State = Player_State_GlideRight
				Player.XVelocity = 0x40000
				Player.Timer = 0
			else
				Player.State = Player_State_GlideLeft
				Player.XVelocity = -0x40000
				Player.Timer = 256
			end if
			Player.Rotation = Player.Direction
			Player.PrevAnimation = ANI_GLIDING
			Player.Animation = ANI_GLIDING		// Couple of extra lines here fix knuckles' front facing frame when he starts a glide lol
			Player.Frame = 2
			Player.AnimationTimer = 0
			Player.AnimationSpeed = 0
		end if
	end if
#endplatform
end function


function Player_State_GlideLeft
#platform: Use_Origins
	if Player.Speed == 0
		if Warp.Destination > 0
			if Warp.Timer > 99
				if Warp.Timer < 204
					Warp.Destination = 0
				end if
			end if
			Warp.Timer = 0
		end if
	end if
	if Player.Gravity == GRAVITY_AIR
		if Player.JumpHold == true

			if Player.Timer == 256
				if Player.Speed < 0x180000
					Player.Speed += 0x400
				end if
			else
				if Player.Speed < 0x40000
					Player.Speed += 0x1000
				end if
			end if

			if Player.YVelocity > 0x8000
				Player.YVelocity -= 0x2000
			else
				Player.YVelocity += 0x2000
			end if
			
			if Player.Timer < 256
				Player.Timer += 4
			end if

			if Player.Timer < 170
				if Player.Timer > 86
					Player.Frame = 0
				else
					if Player.Timer > 44
						Player.Frame = 1
					else
						Player.Frame = 2
					end if
				end if
			else
				if Player.Timer < 212
					Player.Frame = 1
				else
					Player.Frame = 2
				end if
			end if

			if Player.Frame == 2
				TempValue0 = Oscillation
				TempValue0 >>= 3
				TempValue0 %= 2
				if TempValue0 == 1
					Player.Frame = 3
				end if
			end if

			TempValue7 = Player.XPos
			if Player.Timer < 128
				Player.Direction = FACING_RIGHT
				TempValue0 = false
				TempValue1 = false
			else
				Player.Direction = FACING_LEFT
				Player.XPos = TempValue7
				Player.XPos += Player.XVelocity
				Player.YPos = Player.YPos
				Object.XPos = Player.XPos
				Object.YPos = Player.YPos
				ObjectTileCollision(CSIDE_RWALL, -12, -2, Player.CollisionPlane)
				Player.XPos = Object.XPos
				TempValue0 = CheckResult
				TempValue2 = Player.XPos
				Player.XPos = TempValue7
				Player.XPos += Player.XVelocity
				Object.XPos = Player.XPos
				Object.YPos = Player.YPos
				ObjectTileCollision(CSIDE_RWALL, -12, 11, Player.CollisionPlane)
				Player.XPos = Object.XPos
				TempValue1 = CheckResult
				TempValue3 = Player.XPos
			end if
			Cos(Player.XVelocity, Player.Timer)
			Player.XVelocity *= Player.Speed
			Player.XVelocity >>= 9
			if Player.Right == true
				Player.State = Player_State_GlideRight
				if Warp.Destination > 0
					if Warp.Timer > 99
						if Warp.Timer < 204
							Warp.Destination = 0
						end if
					end if
					Warp.Timer = 0
				end if
			end if
			Player.XPos = TempValue7
			CheckResult = TempValue0
			CheckResult &= TempValue1
			if CheckResult == true
				if TempValue2 == TempValue3
					if Player.XVelocity > 0
						Player.Direction = FACING_RIGHT
					end if
					Player.State = Player_State_Climb
					SetBit(Player.DelayLedgePull, 0, true)
					Player.Speed = 0
					Player.XVelocity = 0
					Player.YVelocity = 0
					Player.Timer = 0
					PlaySfx(34, false)	// SFX_G_GRAB
					game.callbackParam0 = 2
					game.callbackParam1 = 0
					game.callbackParam2 = 0
					EngineCallback(NOTIFY_STATS_CHARA_ACTION2)
					if Warp.Destination > 0
						if Warp.Timer > 99
							if Warp.Timer < 204
								Warp.Destination = 0
							end if
						end if
						Warp.Timer = 0
					end if
				else
					if Player.ShouldGlideDrop == 0
						Player.ShouldGlideDrop = 2
					end if
				end if
			else
				if TempValue0 == true
					if TempValue1 == false
						ObjectTileGrip(CSIDE_ENTITY, FACING_LEFT, false, ECEFFECT_NONE)
					end if

					if CheckResult == false
						if Player.ShouldGlideDrop == 0
							Player.ShouldGlideDrop = 2
						end if
					end if
				end if
			end if
		else
			Player.XVelocity >>= 2
			Player.Speed = Player.XVelocity
			TempValue4 = Stage.NewXBoundary1
			TempValue4 += 10
			if Player.iXpos <= TempValue4
				if Player.Direction == FACING_LEFT
					Player.Direction = FACING_RIGHT
				end if
			end if
			CallFunction(Player_Action_GlideDrop)
		end if
	else
		if Player.CollisionMode == CMODE_FLOOR
			Player.Timer = 0
			Player.State = Player_State_GlideSlide
			Player.Animation = ANI_GLIDING_STOP
			Player.Speed = Player.XVelocity
		else
			Player.State = Player_State_Ground
			CallFunction(Player_HandleOnGround)
			Player.Skidding = 0
		end if
	end if

	CallFunction(Player_CheckRoofGlide)
#endplatform
end function

function Player_State_GlideDrop
#platform: Use_Origins
	if Player.Gravity == GRAVITY_AIR
		Player.Timer = 0
		Player.AbilityTimer = 0
		CallFunction(Player_HandleAirFriction)
		CallFunction(Player_HandleAirMovement)
	else
		if Player.JumpPress == true
			Player.XVelocity = 0
			Player.Speed = 0
			if Player.Down == true
				CallFunction(Player_Action_Spindash)
			else
				CallFunction(Player_Action_Jump)
				
				// if you jump from this landing crouch state, you've come to a stop, so consume the current time warp
				// this prevents Knuckles from easily time traveling by just mashing jump after a glide
				if Warp.Timer > 99
					if Warp.Timer < 204
						Warp.Destination = WARPDEST_NONE
					end if
				end if
				Warp.Timer = 0
			end if
		else
			if Player.AbilityTimer == 0
				PlaySfx(33, false)	// SFX_G_GLIDEDROPLAND
			end if

			Player.TrackScroll = false
			Player.Speed = 0
			Player.XVelocity = 0
			Player.Animation = ANI_LOOKINGDOWN
			Player.PrevAnimation = ANI_LOOKINGDOWN
			player.AnimationTimer = 0
			if Player.AbilityTimer > 9
				Player.Frame = 0
			else
				Player.Frame = 1
			end if

			if Player.AbilityTimer < 16
				Player.AbilityTimer++
			else
				Player.AbilityTimer = 0
				Player.Timer = 0
				
				if Player.Up == true
					Player.State = Player_State_LookUp
				else
					if Player.Down == true
						Player.State = Player_State_Crouch
					else
						Player.State = Player_State_Ground
					end if
				end if
				
				CallFunction(Player_HandleOnGround)
				Player.Skidding = 0
				Player.Timer = 0
				if Warp.Destination > 0
					if Warp.Timer > 99
						if Warp.Timer < 204
							Warp.Destination = 0
						end if
					end if
					Warp.Timer = 0
				end if
			end if
		end if
	end if
#endplatform
end function

		if Player.Speed == 0
			if Player.JumpPress == true
				if Player.Down == true
					CallFunction(Player_Action_Spindash)
				else
					CallFunction(Player_Action_Jump)
				end if
			else
				Player.TrackScroll = false
				Player.Frame = 1
				if Player.Timer < 16
					Player.Timer++
				else
					Player.State = Player_State_Ground
					CallFunction(Player_HandleOnGround)
					Player.Skidding = 0
				end if
			end if
		else
			Player.Timer++
			TempValue0 = Player.Timer
			TempValue0 %= 4
			if TempValue0 == 0
				CreateTempObject(TypeName[Dust Puff], 0, Player.XPos, Player.YPos)
				Object[TempObjectPos].iYPos += Player.CollisionBottom
				Object[TempObjectPos].DrawOrder = Player.DrawOrder
			end if

			Player.Timer %= 8
			if Player.Timer == 0
				PlaySfx(32, false)	// SFX_G_SLIDE
			end if
			Player.Frame = 0
			if Player.Speed > 0
				Player.Speed -= 0x2000
				if Player.Speed < 0
					Player.Speed = 0
					Player.Timer = 0
				end if
			else
				Player.Speed += 0x2000
				if Player.Speed > 0
					Player.Speed = 0
					Player.Timer = 0
				end if
			end if
			if Player.JumpHold == 0
				Player.Speed = 0
				Player.Timer = 0
			end if
		end if
		Player.XVelocity = Player.Speed
	else
		CallFunction(Player_Action_GlideDrop)
	end if
#endplatform
end function


function Player_State_Climb
#platform: Use_Origins
	if Player.Gravity == GRAVITY_AIR
		Player.Animation = ANI_CLIMBING
		if Player.Up == true
			if Player.SuperState == SUPERSTATE_SUPER
				Player.YVelocity = -0x20000
			else
				Player.YVelocity = -0x10000
			end if
			
			TempValue0 = Player.CollisionTop
			TempValue0 *= -0x10000
			if Player.YPos < TempValue0
				Player.YPos = TempValue0
			end if
			Player.Timer++
			if Player.Timer == 4
				Player.Timer = 0
				Player.Frame++
				if Mini_PlayerFlag == true
					Player.Frame %= 2
				else
					Player.Frame %= 6
				end if
			end if
		else
			if Player.Down == true
				if Player.SuperState == SUPERSTATE_SUPER
					Player.YVelocity = 0x20000
				else
					Player.YVelocity = 0x10000
				end if
				
				Player.Timer++
				if Player.Timer == 4
					Player.Timer = 0
					if Player.Frame < 1
						if Mini_PlayerFlag == true
							Player.Frame += 2
						else
							Player.Frame += 6
						end if
					end if
					Player.Frame--
				end if
			else
				Player.YVelocity = 0
			end if
		end if
		if Player.JumpPress == true
			Player.Animation = ANI_JUMPING
			Player.State = Player_State_Air_NoDropDash
			Player.Timer = 0

			PlaySfx(0, false)	// SFX_G_JUMP
			if Player.Direction == FACING_LEFT
				Player.XVelocity = 0x40000
				Player.Speed = 0x40000
				Player.Direction = FACING_RIGHT
			else
				Player.XVelocity = -0x40000
				Player.Speed = -0x40000
				Player.Direction = FACING_LEFT
			end if
			Player.YVelocity = -0x40000
			if Player.GravityStrength != 0x3800 // (see if Knuckles is underwater)
				Player.XVelocity >>= 1
				Player.Speed >>= 1
				Player.YVelocity >>= 1
			end if
		else
			if Mini_PlayerFlag == true
				TempValue4 = 7
				TempValue5 = -6
			else
				TempValue4 = 10
				TempValue5 = -10
			end if

			if Player.Direction == FACING_RIGHT
				TempValue2 = Player.XPos
				Object.XPos = Player.XPos
				Object.YPos = Player.YPos
				ObjectTileGrip(CSIDE_LWALL, TempValue4, TempValue5, Player.CollisionPlane)
				TempValue0 = CheckResult
				if TempValue0 == 0
					Player.XPos = TempValue2
					Object.XPos = Player.XPos
					ObjectTileGrip(CSIDE_LWALL, TempValue4, 11, Player.CollisionPlane)
					ObjectTileGrip(CSIDE_ENTITY, FACING_RIGHT, true, ECEFFECT_NONE)
					TempValue0 = CheckResult
				end if
				Player.XPos = Object.XPos
				TempValue3 = Player.XPos
				Player.XPos = TempValue2
				Object.XPos = Player.XPos
				Object.YPos = Player.YPos
				ObjectTileGrip(CSIDE_LWALL, TempValue4, 11, Player.CollisionPlane)
				Player.XPos = Object.XPos
				TempValue1 = CheckResult
				if TempValue1 == 0
					ObjectTileGrip(CSIDE_ENTITY, FACING_RIGHT, false, ECEFFECT_NONE)
					TempValue1 = CheckResult
				end if
				if Player.XPos > TempValue3
					Player.XPos = TempValue3
				end if
				ObjectTileGrip(CSIDE_ENTITY, FACING_RIGHT, false, ECEFFECT_RESETSTORAGE)
			else
				if Mini_PlayerFlag == true
					TempValue4 = -7
				else
					TempValue4 = -10
				end if

				TempValue2 = Player.XPos
				Object.XPos = Player.XPos
				Object.YPos = Player.YPos
				ObjectTileGrip(CSIDE_RWALL, TempValue4, TempValue5, Player.CollisionPlane)
				TempValue0 = CheckResult
				if TempValue0 == 0
					Player.XPos = TempValue2
					Object.XPos = Player.XPos
					ObjectTileGrip(CSIDE_RWALL, TempValue4, 11, Player.CollisionPlane)
					ObjectTileGrip(CSIDE_ENTITY, FACING_LEFT, true, ECEFFECT_NONE)
					TempValue0 = CheckResult
				end if
				Player.XPos = Object.XPos
				TempValue3 = Player.XPos
				Player.XPos = TempValue2
				Object.XPos = Player.XPos
				Object.YPos = Player.YPos
				ObjectTileGrip(CSIDE_RWALL, TempValue4, 11, Player.CollisionPlane)
				Player.XPos = Object.XPos
				TempValue1 = CheckResult
				if TempValue1 == 0
					ObjectTileGrip(CSIDE_ENTITY, FACING_LEFT, false, ECEFFECT_NONE)
					TempValue1 = CheckResult
				end if
				if Player.XPos < TempValue3
					Player.XPos = TempValue3
				end if
				ObjectTileGrip(CSIDE_ENTITY, FACING_LEFT, false, ECEFFECT_RESETSTORAGE)
			end if
			if TempValue0 == 0
				GetBit(TempValue0, Player.DelayLedgePull, 0)
				if TempValue0 == false
					// fix a weird bug where if the player's subpixel is exactly 0, they'll be placed one pixel too low
					// this frequently happens when climbing blocks, somehow they must be clearing the subpixel...
					TempValue0 = Player.YPos
					TempValue0 &= 0xFFFF
					if TempValue0 == 0
						Player.YPos--		// nudging this back up by only one subpixel is sufficient
					end if
					
					Player.XPos = TempValue2
					Player.Animation = ANI_LEDGEPULLUP
					Player.YVelocity = 0
					Player.Timer = 0
					Player.State = Player_State_LedgePullUp
					
					Object.XPos = Player.XPos
					Object.YPos = Player.YPos
					if Player.Direction == FACING_RIGHT
						ObjectTileGrip(CSIDE_FLOOR, 12, -9, Player.CollisionPlane)
					else
						ObjectTileGrip(CSIDE_FLOOR, -12, -9, Player.CollisionPlane)
					end if
					if CheckResult == true
						Player.YPos = Object.YPos	// this was missing in vanilla Origins, resulting in the ObjectTileGrip doing NOTHING! :D
					end if
					
					Player.TileCollisions = false
					if Player.Direction == FACING_RIGHT
						Player.XPos += 0x10000
					else
						Player.XPos -= 0x10000
					end if
				else
					SetBit(Player.DelayLedgePull, 1, true)
				end if
			else
				if TempValue1 == 0
					Player.PrevAnimation = ANI_GLIDING_DROP
					Player.Frame = 2
					CallFunction(Player_Action_GlideDrop)
				end if
			end if
		end if
	else
		Player.Animation = ANI_WALKING
		Player.State = Player_State_Ground
		CallFunction(Player_HandleOnGround)
	end if
	
	SetBit(Player.DelayLedgePull, 0, false)
#endplatform
end function


function Player_State_LedgePullUp
#platform: Use_Origins
	switch Player.Frame
	case 0
		if Player.Timer < 5
			Object.XPos = Player.XPos
			Object.YPos = Player.YPos
			if Player.Direction == FACING_RIGHT
				ObjectTileGrip(CSIDE_FLOOR, 12, -9, Player.CollisionPlane)
			else
				ObjectTileGrip(CSIDE_FLOOR, -12, -9, Player.CollisionPlane)
			end if
			if CheckResult == true
				Player.YPos = Object.YPos	// this was missing in vanilla Origins, resulting in the ObjectTileGrip doing NOTHING! :D
			end if
			
			Player.Timer++
		else
			Player.Timer = 0
			Player.Frame++
			if Player.Direction == FACING_RIGHT
				Player.XPos += 0x90000
			else
				Player.XPos -= 0x90000
			end if
			Player.YPos -= 0xA0000
		end if
		break

	case 1
		if Player.Timer < 5
			Player.Timer++
		else
			Player.Timer = 0
			Player.Frame++
			if Player.Direction == FACING_RIGHT
				Player.XPos += 0x40000
			else
				Player.XPos -= 0x40000
			end if
			
			if Mini_PlayerFlag == false
				Player.YPos -= 0x20000
			end if
		end if
		break
	case 2
		if Player.Timer < 5
			Player.Timer++
		else
			Player.Timer = 0
			Player.Animation = ANI_STOPPED
			Player.State = Player_State_Air_NoDropDash
			Player.YPos -= 0xA0000
			Player.TileCollisions = true
		end if
		break
	end switch
#endplatform
end function


function Player_State_GlideLeftNoGrip
#platform: Use_Origins
	if Player.Gravity == GRAVITY_AIR
		if Player.JumpHold == true
		
			if Player.Timer == 256
				if Player.Speed < 0x180000
					Player.Speed += 0x400
				end if
			else
				if Player.Speed < 0x40000
					Player.Speed += 0x1000
				end if
			end if

			if Player.YVelocity > 0x8000
				Player.YVelocity -= 0x2000
			else
				Player.YVelocity += 0x2000
			end if
			
			if Player.Timer < 256
				Player.Timer += 4
			end if

			if Player.Timer < 170
				if Player.Timer > 86
					Player.Frame = 0
				else
					if Player.Timer > 44
						Player.Frame = 1
					else
						Player.Frame = 2
					end if
				end if
			else
				if Player.Timer < 212
					Player.Frame = 1
				else
					Player.Frame = 2
				end if
			end if

			if Player.Frame == 2
				TempValue0 = Oscillation
				TempValue0 >>= 3
				TempValue0 %= 2
				if TempValue0 == 1
					Player.Frame = 3
				end if
			end if

			TempValue7 = Player.XPos
			if Player.Timer < 128
				Player.Direction = FACING_RIGHT
				TempValue0 = false
				TempValue1 = false
			else
				Player.Direction = FACING_LEFT
				Player.XPos = TempValue7
				Player.XPos += Player.XVelocity
				Player.YPos = Player.YPos
				Object.XPos = Player.XPos
				Object.YPos = Player.YPos
				ObjectTileCollision(CSIDE_RWALL, -12, -2, Player.CollisionPlane)
				Player.XPos = Object.XPos
				TempValue0 = CheckResult
				TempValue2 = Player.XPos
				Player.XPos = TempValue7
				Player.XPos += Player.XVelocity
				Object.XPos = Player.XPos
				Object.YPos = Player.YPos
				ObjectTileCollision(CSIDE_RWALL, -12, 11, Player.CollisionPlane)
				Player.XPos = Object.XPos
				TempValue1 = CheckResult
				TempValue3 = Player.XPos
			end if
			Cos(Player.XVelocity, Player.Timer)
			Player.XVelocity *= Player.Speed
			Player.XVelocity >>= 9
			if Player.Right == true
				Player.State = Player_State_GlideRightNoGrip
			end if
			Player.XPos = TempValue7
			CheckResult = TempValue0
			CheckResult &= TempValue1
			if CheckResult == true
				// Dev note: The following code doesn't seem like it would actually work, so I'm removing it for efficiency
				//TempValue0 = false
				//ArrayPos0 = 32
				//while ArrayPos0 < 1056
				//	CheckEqual(Object[ArrayPos0].Type, TypeName[Blank Object]) // huh? maybe this is supposed to be [NoGripArea], but you can't use stage object TypeNames in global objects...
				//	if CheckResult == true
				//		if Object[ArrayPos0].Value0 == true
				//			TempValue0 = true
				//		end if
				//	end if
				//	ArrayPos0++
				//loop

				//if TempValue0 == false
					if TempValue2 != TempValue3
						if Player.ShouldGlideDrop == 0
							Player.ShouldGlideDrop = 2
						end if
					end if
				//end if
			end if
		else
			Player.XVelocity >>= 2
			Player.Speed = Player.XVelocity
			TempValue4 = Stage.NewXBoundary1
			TempValue4 += 10
			if Player.iXpos <= TempValue4
				if Player.Direction == FACING_LEFT
					Player.Direction = FACING_RIGHT
				end if
			end if
			CallFunction(Player_Action_GlideDrop)
		end if
	else
		if Player.CollisionMode == CMODE_FLOOR
			Player.Timer = 0
			Player.State = Player_State_GlideSlide
			Player.Animation = ANI_GLIDING_STOP
			Player.Speed = Player.XVelocity
		else
			Player.State = Player_State_Ground
			CallFunction(Player_HandleOnGround)
			Player.Skidding = 0
		end if
	end if

	CallFunction(Player_CheckRoofGlide)
#endplatform
end function

function Player_CheckRoofGlide
#platform: Use_Origins
// Prevent Knuckles from gliding over stages at all.
	TempValue0 = Stage.YBoundary1
	TempValue0 += 16
	TempValue0 <<= 16
	if Player.YPos < TempValue0
		Player.XVelocity = 0
		Player.Speed = Player.XVelocity
	end if
#endplatform
end function


function Player_Action_GlideDrop
#platform: Use_Origins
	// Bug Details:
	// Player.Timer should probably be cleared here, but it isn't
	// Because of this, doing things like landing from a drop after a left-facing glide will make Knuckles skip to the middle of his idle animation immediately
	// It's fixed here in this mod.
	Player.Timer = 0
	Player.Animation = ANI_GLIDING_DROP
	Player.AbilityTimer = 0
	Player.State = Player_State_GlideDrop
#endplatform
end function


function Player_Action_DblJumpAmy
#platform: Use_Origins
	// Handle the super activation
	GetBit(CheckResult, secrets.gameplay, SECRET_GP_SUPERFORMS)
	if CheckResult == true
		if KeyPress.ButtonY != false
			// Following checks are to see if Amy should transform instead of doing arial abilities

			// Each bit represents an emerald's collected state - 0x7F is 7 1's, AKA 7 emeralds
			CheckEqual(SpecialStage.TimeStones, 0x7F)
			TempValue0 = CheckResult

			// Player needs to have at least 50 rings
			CheckGreater(Player.Rings, 49)
			TempValue0 &= CheckResult

			// Player can't already be Super, of course
			CheckNotEqual(Player.SuperState, SUPERSTATE_SUPER)
			TempValue0 &= CheckResult

			// Don't let the player transform if the level is over, either
			CheckNotEqual(Object[30].Type, TypeName[ActFinish])
			TempValue0 &= CheckResult
			
			if TempValue0 == true
				// All checks passed, transform!
				CallFunction(Player_TryTransform)
			end if
		end if
	end if

	if Player.JumpPress == true
		CheckNotEqual(Player.State, Player_State_Transform) // listen... this is the only way i could get this shit to work since the transform key is always a jump press. :sobs:
		if CheckResult == true
			PlaySfx(29, false)	// SFX_G_HAMMERJUMP
			game.callbackParam0 = 0
			game.callbackParam1 = 1
			game.callbackParam2 = 0
			EngineCallback(NOTIFY_STATS_CHARA_ACTION2)
			
			// Change the animation without resetting the frame or animation speed
			// Bug Details: The way Sega did this causes the frame timer to reset, leaving the current frame on longer than intended
			// Idk why they even bothered with any of this though since the hammer jump frames don't line up anyway?
			TempValue0 = Player.AnimationSpeed
			TempValue1 = Player.Frame
			Player.Animation = ANI_HAMMER_JUMP
			ProcessAnimation()
			Player.AnimationSpeed = TempValue0
			
			// adjust frame to match hammer jump frame
			if TempValue1 == 4
				TempValue1 = 7
			else
				TempValue1 <<= 1
			end if
			
			Player.Frame = TempValue1
			
	//		// This is how you do it the proper way btw
	//		Player.Animation = ANI_HAMMER_JUMP
	//		Player.PrevAnimation = ANI_HAMMER_JUMP
	//		
	//		// adjust frame to match hammer jump frame
	//		if Player.Frame == 4
	//			Player.Frame = 7
	//		else
	//			Player.Frame <<= 1
	//		end if
		end if
	end if
#endplatform
end function


// insta-shield/elemental shields
function Player_Action_DblJumpSonic
#platform: Use_Origins
	// Handle the super activation
	GetBit(CheckResult, secrets.gameplay, SECRET_GP_SUPERFORMS)
	if CheckResult == true
		if KeyPress.ButtonY != false // fun fact: this is a global variable tied to the input system. Only for Sonic CD. :steamhappy:
			// Following checks are to see if Sonic should transform instead of doing arial abilities

			// Each bit represents an emerald's collected state - 0x7F is 7 1's, AKA 7 emeralds
			CheckEqual(SpecialStage.TimeStones, 0x7F)
			TempValue0 = CheckResult

			// Player needs to have at least 50 rings
			CheckGreater(Player.Rings, 49)
			TempValue0 &= CheckResult

			// Player can't already be Super, of course
			CheckNotEqual(Player.SuperState, SUPERSTATE_SUPER)
			TempValue0 &= CheckResult

			// Don't let the player transform if the level is over, either
			CheckNotEqual(Object[30].Type, TypeName[ActFinish])
			TempValue0 &= CheckResult
			
			if TempValue0 == true
				// All checks passed, transform!
				CallFunction(Player_TryTransform)
				game.callbackParam0 = 1
				game.callbackParam1 = 0
				game.callbackParam2 = 0
				EngineCallback(NOTIFY_STATS_CHARA_ACTION)
			end if
		end if
	end if
	
	if Player.JumpPress == true
		CheckNotEqual(Player.State, Player_State_Transform) // listen... this is the only way i could get this shit to work since the transform key is always a jump press. :sobs:
		if CheckResult == true
			if Player.SuperState != SUPERSTATE_SUPER
				if Object[+2].Type == TypeName[Blank Object]
					GetBit(CheckResult, secrets.gameplay, SECRET_GP_SHIELDS)
					if CheckResult == true
						PlaySfx(35, false) // SFX_G_INSTASHIELD
						
						TempValue0 = Player.EntityNo
						TempValue0 += 2
						ResetObjectEntity(TempValue0, TypeName[Blue Shield], NO_SHIELD, Player.XPos, Player.YPos)
						Object[+2].ShieldType		= SHIELD_INSTA
						Object[+2].Priority			= PRIORITY_ACTIVE
						
						Player.JumpAbilityState = 2
					end if
				else
					if Object[+2].Type == TypeName[Blue Shield]
						switch Object[+2].ShieldType
						
						case SHIELD_BUBBLE
							// Start the bubble bounce
							PlaySfx(41, false) // SFX_G_BUBBLEBOUNCE

							Player.YVelocity = 0x80000
							Player.XVelocity = 0
							Player.Speed = 0
							Player.JumpAbilityState = 2
							Player.State = Player_State_BubbleBounce

							// Set the bubble shield state to falling
							Object[+2].Animation = 1	// SHIELD_ANIM_BUBBLE_D
							break
						
						case SHIELD_FIRE
							PlaySfx(37, false) // SFX_G_FIREDASH

							// Determine launch direction based on where the player's currently facing
							GetBit(TempValue0, Player.Direction, 0)

							if TempValue0 == false
								Player.XVelocity = 0x80000
							else
								Player.XVelocity = -0x80000
							end if

							Player.Speed = Player.XVelocity
							Player.YVelocity = 0
							Player.JumpAbilityState = 2

							// Lock the camera momentarily
							if Options.OriginalControls == true
								Screen.CameraStyle = CAMERASTYLE_EXTENDED
							else
								Player.ScrollDelay = 15
								Screen.CameraStyle = CAMERASTYLE_HLOCKED
							end if

							// Set the flame shield to active and make it match the player's direction
							Object[+2].Animation = 4	// SHIELD_ANIM_FIRE_DASH

// Function that transforms the player
function Player_TryTransform
	// Only allow the player to transform during the actual level, not when results are showing or other blocking elements
	if Stage.TimeEnabled == true
		// Spawn the sparkly super trail object
		ArrayPos0  = Player.EntityNo
		ArrayPos0 += 2
		TempValue0 = Object[ArrayPos0].PropertyValue
		ResetObjectEntity(ArrayPos0, TypeName[Dust Puff], TempValue0, Player.XPos, Player.YPos)
		Object[ArrayPos0].Value3	= true	// mark it as the spark controller
		Object[ArrayPos0].Frame		= 30	// set the spark frame
		Object[ArrayPos0].Priority	= PRIORITY_ACTIVE

		
		GetBit(checkResult, secrets.cosmetic, 1) // SECRET_CM_NOSUPERMUSIC
		if checkResult != 1
			if Music.CurrentTrack != 8
				if Music.CurrentTrack != 4	// Boss music
					PlayMusic(8)	// Super music
				end if
			end if
		end if
		
		if Player.SuperState != SUPERSTATE_QUEUED
			Player.SuperState = SUPERSTATE_SUPER
		end if

		Player.InvincibleTimer = 60

		CallFunction(Player_UpdatePhysicsState)

		// Load Super Sonic sprites if Sonic
		if Player.SuperState != SUPERSTATE_QUEUED
			if Stage.PlayerListPos == PLAYER_SONIC_A
				if Mini_PlayerFlag == false
					LoadAnimation("SuperSonic.Ani")
					BindPlayerToObject(0, 0)
				end if
			end if
		end if
		
		// fix momentum going the wrong way if the player was gliding after breaking an S monitor
		CheckEqual(Player.State, Player_State_GlideLeft)
		TempValue0 = CheckResult
		CheckEqual(Player.State, Player_State_GlideRight)
		TempValue0 |= CheckResult
		CheckEqual(Player.State, Player_State_GlideLeftNoGrip)
		TempValue0 |= CheckResult
		CheckEqual(Player.State, Player_State_GlideRightNoGrip)
		TempValue0 |= CheckResult
		if TempValue0 == true
			if Player.Direction == FACING_LEFT
				FlipSign(Player.Speed)
			end if
		end if
		
		if Player.Gravity == GRAVITY_GROUND	// we love Sonic 3 quirks
			Player.Speed = 0
			Player.XVelocity = 0
		end if

		Player.State = Player_State_Transform
		Player.OffsetJump = true
		if Mini_PlayerFlag == false
			Player.Animation = 33	// ANI_SUPER_TRANSFORM
			if Player.PrevAnimation == 33	// ANI_SUPER_TRANSFORM
				Player.PrevAnimation = 0	// force transform anim to restart
			end if
		end if
		Player.Timer = 24
		
		// did you know? S3K does this!
		Player.AirTimer = 0
		//Player.AirLevel = 0	// this is handled by the State of the Water object in CD, though somehow it's automatically being reset anyway, so whatevs
		
		// make sure these are cleared from last time (if there was a last time)
		Player.SuperRingLossTimer = 0
		Player.SuperBlendClr = 0
		Player.SuperBlendTimer = 0
	end if
end function

// Temporary state for when the player is in their transform animation
function Player_State_Transform
	// Delay is normally 24 frames, set in the transform script rather than here
	Player.Timer--

	if Player.Timer == 11 // this mimics when the transform sound plays like S3K does.
		PlaySfx(43, false) // SFX_G_TRANSFORM
	end if
	
	if Player.Timer >= 0
		Player.OffsetJump = true
	else
		if Player.Timer == -15
			Player.Animation = ANI_WALKING
			Player.Timer = 0
			Player.State = Player_State_Air_NoDropDash
		end if
		CallFunction(Player_State_Air)
		
		// make sure the timer never lingers as a negative number
		// this may be unnecessary but better off not taking chances
		if Player.State != Player_State_Transform
			if Player.Timer < 0
				Player.Timer = 0
			end if
		end if
	end if
end function


function Player_HandleSuperForm
	CheckEqual(Player.SuperState, SUPERSTATE_UNTRANSFORM)	// check if the player was detransformed by another object (signposts and capsules) FIRST
	TempValue2 = CheckResult								// this catches a potential edge case where the palette cycle was right at the beginning, failing to restore the normal physics and etc.

	// Do the fancy palette stuff if Super or while untransforming back to normal
	if Player.SuperState != SUPERSTATE_NONE
		switch Stage.PlayerListPos
		case PLAYER_SONIC_A
		//case PLAYER_SONIC_TAILS_A // This case check should never pass since playing as S&T will set your character to Sonic alone
			CallFunction(Player_HandleSuperPalette_Sonic)
			break

		case PLAYER_TAILS_A
			CallFunction(Player_HandleSuperPalette_Tails)
			break

		case PLAYER_KNUCKLES_A
			CallFunction(Player_HandleSuperPalette_Knux)
			break

		case PLAYER_AMY_A
			CallFunction(Player_HandleSuperPalette_Amy)
			break
		end switch
	end if

	if Player.SuperState == SUPERSTATE_SUPER
		Player.InvincibleTimer = 60

		Player.SuperRingLossTimer++

		// If a second has passed since the last ring loss...
		if Player.SuperRingLossTimer == 60
			// ...then take another ring from the player
			Player.SuperRingLossTimer = 0
			Player.Rings--

			// If the player has no more rings, make them untransform
			if Player.Rings == 0
				Player.SuperState = SUPERSTATE_UNTRANSFORM
			end if
		end if
	end if

	CheckEqual(Player.SuperState, SUPERSTATE_UNTRANSFORM)	// check if the player is detransforming, or WAS detransforming when entering this function (see above)
	TempValue2 |= checkResult								// otherwise there's a chance the super palette function already set it to SUPERSTATE_NONE, failing to run this code
	if TempValue2 == true
		if Stage.PlayerListPos == PLAYER_SONIC_A
			if Mini_PlayerFlag == false
				LoadAnimation("Sonic.ani")
				BindPlayerToObject(0, 0)
				if Player.Animation == ANI_WAITING
					Player.Animation = ANI_STOPPED
					Player.Frame = 0
				end if
			end if
		end if

		if Music.CurrentTrack == 8
			PlayMusic(0)
		end if

		Player.InvincibleTimer = 0

		// If the player isn't already dead, give them back their shield
		if Player.State != Player_State_Death
			if Player.State != Player_State_Drown
				switch Object[+2].PropertyValue
				case NO_SHIELD
					TempValue0  = Player.EntityNo
					TempValue0 += 2
					ResetObjectEntity(TempValue0, TypeName[Blank Object], 0, 0, 0)
					break

				case ACTIVE_SHIELD
					TempValue0 = Player.EntityNo
					TempValue0 += 2
					ResetObjectEntity(TempValue0, TypeName[Blue Shield], ACTIVE_SHIELD, Player.XPos, Player.YPos)
					Object[+2].ShieldType		= SHIELD_NORMAL
					Object[+2].InkEffect		= 2 // INK_ALPHA
					Object[+2].Alpha			= 160
					Object[+2].Priority			= PRIORITY_ACTIVE
					break
					
				case SHIELD_BUBBLE
					TempValue0 = Player.EntityNo
					TempValue0 += 2
					ResetObjectEntity(TempValue0, TypeName[Blue Shield], SHIELD_BUBBLE, Player.XPos, Player.YPos)
					Object[+2].ShieldType		= SHIELD_BUBBLE
					Object[+2].Animation		= 0	// SHIELD_ANIM_BUBBLE
					Object[+2].Priority			= PRIORITY_ACTIVE
					break
					
				case SHIELD_FIRE
					TempValue0 = Player.EntityNo
					TempValue0 += 2
					ResetObjectEntity(TempValue0, TypeName[Blue Shield], SHIELD_FIRE, Player.XPos, Player.YPos)
					Object[+2].ShieldType		= SHIELD_FIRE
					Object[+2].Animation		= 3	// SHIELD_ANIM_FIRE
					Object[+2].Priority			= PRIORITY_ACTIVE
					break
					
				case SHIELD_LIGHTNING
					TempValue0 = Player.EntityNo
					TempValue0 += 2
					ResetObjectEntity(TempValue0, TypeName[Blue Shield], SHIELD_LIGHTNING, Player.XPos, Player.YPos)
					Object[+2].ShieldType		= SHIELD_LIGHTNING
					Object[+2].Animation		= 5	// SHIELD_ANIM_LIGHTNING
					Object[+2].Priority			= PRIORITY_ACTIVE
					break

				end switch
			end if
		end if

		if Player.SuperState == SUPERSTATE_UNTRANSFORM
			Player.SuperState = SUPERSTATE_END
		end if
		CallFunction(Player_UpdatePhysicsState)
	end if
end function


function Player_HandleSuperPalette_Sonic
	if Player.SuperState == SUPERSTATE_SUPER
		TempValue0 = false
		if Player.Animation == 33	// ANI_SUPER_TRANSFORM
			if Player.Timer < 13	// delay to match S3K (note: technically this is wrong, it should actually be 11 to match mathemetically, but S1/2's fade-in colors last for 4 frames instead of 2, so it synchronizes better like this)
				TempValue0 = true
			end if
		else
			TempValue0 = true
		end if
		
		if TempValue0 == true
			Player.SuperBlendTimer++
			if Player.SuperBlendTimer >= 4
				Player.SuperBlendTimer = 0

				Player.SuperBlendClr += 4
				if Player.SuperBlendClr >= 64
					Player.SuperBlendClr = 24
				end if
			end if
		end if
	else
		Player.SuperBlendTimer++
		if Player.SuperBlendTimer >= 8
			Player.SuperBlendTimer = 0

			Player.SuperBlendClr -= 4
			if Player.SuperBlendClr <= 0
				Player.SuperBlendClr = 0
				Player.SuperState = SUPERSTATE_NONE
				
				if Player.Animation == ANI_STOPPED
					if Player.Timer == 240
						Player.Animation = ANI_WAITING
					end if
				end if
			end if

			if Player.SuperBlendClr >= 24
				Player.SuperBlendClr = 24
			end if
		end if
	end if

	// set the actual palette
	TempValue0 = Player.SuperBlendClr		// take the palette index
	TempValue0 += 1536						// move to bank 6, where the Super palette cycle is stored
	
	game.callbackParam0 = TempValue0		// Src Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
	game.callbackParam2 = 4					// Count of how many palette slots are going to get copied.
	
	// loop to copy it into banks 0-5
	TempValue1 = 39							// Dst Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
	CheckNotEqual(Stage.ListPos, 13) // and are we not in CCZ1D?
	TempValue7 = CheckResult
	CheckNotEqual(Stage.ListPos, 17) // and are we not in CCZ2D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 19) // and are we not in CCZ3D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 81) // and are we not in ANCCZ1D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 85) // and are we not in ANCCZ2D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 87) // and are we not in ANCCZ3D?
	TempValue7 &= CheckResult
	if TempValue7 == true // if we arent in either of these...
		while TempValue1 < 1536
			game.callbackParam1 = TempValue1
			EngineCallback(CALLBACK_COPYPALETTE)	// Call the new CopyPalette feature!
			
			TempValue1 += 256
		loop
	else // if we are in either of these stages
		game.callbackParam1 = TempValue1
		EngineCallback(CALLBACK_COPYPALETTE) // just copy to Bank 0
	end if
	
	if HUD.IsTidalTempest == true
		game.callbackParam0 += 64				// Src Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
		game.callbackParam1 = 1831				// Dst Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
		EngineCallback(CALLBACK_COPYPALETTE)	// Call the new CopyPalette feature!
	end if
end function


function Player_HandleSuperPalette_Tails
	if Player.SuperState == SUPERSTATE_SUPER
		TempValue0 = false
		if Player.Animation == 33	// ANI_SUPER_TRANSFORM
			if Player.Timer < 12	// delay to match S3K
				TempValue0 = true
			end if
		else
			TempValue0 = true
		end if
		
		if TempValue0 == true
			// Main Super state
			Player.SuperBlendTimer++
			if Player.SuperBlendTimer >= 12
				Player.SuperBlendTimer = 0
				Player.SuperBlendClr += 4

				if Player.SuperBlendClr >= 24
					Player.SuperBlendClr = 0
				end if
			end if
		end if
	else
		// Detransforming
		if Player.SuperBlendClr > 12
			FlipSign(Player.SuperBlendClr)
			Player.SuperBlendClr += 24
		end if

		Player.SuperBlendTimer++
		if Player.SuperBlendTimer >= 8
			Player.SuperBlendTimer = 0
			Player.SuperBlendClr -= 4

			if Player.SuperBlendClr <= 0
				Player.SuperBlendClr = 0
				Player.SuperState = SUPERSTATE_NONE
			end if
		end if
	end if
	
	// set the actual palette
	TempValue0 = Player.SuperBlendClr		// take the palette index
	TempValue0 += 1536						// move to bank 6, where the Super palette cycle is stored
	
	game.callbackParam0 = TempValue0		// Src Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
	game.callbackParam2 = 4					// Count of how many palette slots are going to get copied.
	
	// loop to copy it into banks 0-5
	TempValue1 = 44							// Dst Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
	CheckNotEqual(Stage.ListPos, 13) // and are we not in CCZ1D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 17) // and are we not in CCZ2D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 19) // and are we not in CCZ3D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 81) // and are we not in ANCCZ1D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 85) // and are we not in ANCCZ2D?
	TempValue7 &= CheckResult
	CheckNotEqual(Stage.ListPos, 87) // and are we not in ANCCZ3D?
	TempValue7 &= CheckResult
	if TempValue7 == true // if we arent in either of these...
		while TempValue1 < 1536
			game.callbackParam1 = TempValue1
			EngineCallback(CALLBACK_COPYPALETTE)	// Call the new CopyPalette feature!
			
			TempValue1 += 256
		loop
	else // if we are in either of these stages
		game.callbackParam1 = TempValue1
		EngineCallback(CALLBACK_COPYPALETTE) // just copy to Bank 0
	end if
	
	if HUD.IsTidalTempest == true
		game.callbackParam0 += 64				// Src Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
		game.callbackParam1 = 1836				// Dst Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
		EngineCallback(CALLBACK_COPYPALETTE)	// Call the new CopyPalette feature!
	end if
end function


function Player_HandleSuperPalette_Knux
	if Player.SuperState == SUPERSTATE_SUPER
		TempValue0 = false
		if Player.Animation == 33	// ANI_SUPER_TRANSFORM
			if Player.Timer < 21	// delay to match S3K (this one starts sooner because the cycle is different in 3K I think, so we need to account for that)
				TempValue0 = true
			end if
		else
			TempValue0 = true
		end if
		
		if TempValue0 == true
			// Main Super state
			Player.SuperBlendTimer++
			if Player.SuperBlendTimer >= 12
				Player.SuperBlendTimer = 9
				Player.SuperBlendClr += 4

				if Player.SuperBlendClr >= 44
					Player.SuperBlendTimer = 0
					Player.SuperBlendClr = 4
				end if
			end if
		end if
	else
		// Detransforming
		if Player.SuperBlendClr > 24
			Player.SuperBlendClr -= 4
			FlipSign(Player.SuperBlendClr)
			Player.SuperBlendClr += 44
		end if

		Player.SuperBlendTimer++
		if Player.SuperBlendTimer >= 8
			Player.SuperBlendTimer = 0
			Player.SuperBlendClr -= 4

			if Player.SuperBlendClr <= 0
				Player.SuperBlendClr = 0
				Player.SuperState = SUPERSTATE_NONE
			end if
		end if
	end if
	
	// set the actual palette
	TempValue0 = Player.SuperBlendClr		// take the palette index
	TempValue0 += 1536						// move to bank 6, where the Super palette cycle is stored
	
	game.callbackParam0 = TempValue0		// Src Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
	game.callbackParam2 = 4					// Count of how many palette slots are going to get copied.
	
	// loop to copy it into banks 0-5
	TempValue1 = 76							// Dst Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
	while TempValue1 < 1536
		game.callbackParam1 = TempValue1
		EngineCallback(CALLBACK_COPYPALETTE)	// Call the new CopyPalette feature!
		
		TempValue1 += 256
	loop
	
	if HUD.IsTidalTempest == true
		game.callbackParam0 += 64				// Src Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
		game.callbackParam1 = 1868				// Dst Palette, which is base index number and can have 256 (in multiple) added onto it to go into different banks.
		EngineCallback(CALLBACK_COPYPALETTE)	// Call the new CopyPalette feature!
	end if
end function

// When the player is falling from a bubble bounce
function Player_State_BubbleBounce
	CallFunction(Player_HandleAirFriction)
	
	// Check if the player should stop bubble-bouncing
	CheckNotEqual(Object[+2].ShieldType, SHIELD_BUBBLE)
	TempValue0 = CheckResult
	
	CheckEqual(Player.SuperState, SUPERSTATE_SUPER)
	TempValue0 |= CheckResult
	
	if TempValue0 == true
		// Go back to normal air falling
#platform: Use_Standalone
		Player.State = Player_State_Air
#endplatform
#platform: Use_Origins
		Player.State = Player_State_Air_NoDropDash
#endplatform
	else
		if Player.Gravity == GRAVITY_AIR
			CallFunction(Player_HandleAirMovement)
		else
			// Allow the player to bounce again after rebound
			Player.JumpAbilityState = 1

			Player.Gravity = GRAVITY_AIR

			// Underwater?
			if Player.JumpStrength == 0x38000
				TempValue1 = 0x40000
			else
				TempValue1 = 0x78000
			end if

			TempValue1 += Player.GravityStrength

			Sin256(Player.XVelocity, Player.Angle)
			Player.XVelocity *= TempValue1

			Cos256(TempValue0, Player.Angle)
			TempValue0 *= Player.Speed

			Player.XVelocity += TempValue0
			Player.XVelocity >>= 8

			Sin256(Player.YVelocity, Player.Angle)
			Player.YVelocity *= Player.Speed

			Cos256(TempValue0, Player.Angle)
			TempValue0 *= TempValue1

			Player.YVelocity -= TempValue0
			Player.YVelocity >>= 8

			Player.Speed = Player.XVelocity

			//player.scrollTracking = true
			Player.Animation = ANI_JUMPING

			Player.Angle = 0
			Player.CollisionMode = CMODE_FLOOR
			Player.Timer = 1

			CallFunction(Player_HandleRollAnimSpeed)

#platform: Use_Standalone
			Player.State = Player_State_Air
#endplatform
#platform: Use_Origins
			Player.State = Player_State_Air_NoDropDash
#endplatform

			PlaySfx(41, false) // SFX_G_BUBBLEBOUNCE

			// Set the bubble shield bounce animation
			Object[+2].Animation = 2	// SHIELD_ANIM_BUBBLE_U
			
			// no cheese
			TempValue0 = Player.XVelocity
			if TempValue0 < 0
				FlipSign(TempValue0)
			end if
			if TempValue0 < 0x10000
				if Warp.Timer > 99
					if Warp.Timer < 204
						Warp.Destination = WARPDEST_NONE
					end if
				end if
				Warp.Timer = 0
			end if
		end if
	end if
end function


function Player_Action_HammerDash
#platform: Use_Origins
	StopSfx(28)	// SFX_G_DROPDASH
	PlaySfx(30, false)	// SFX_G_HAMMERDASH
	Player.State = Player_State_HammerDash
	Player.Animation = ANI_HAMMER_DASH
	Player.Timer = 0
	Player.DropDashCharge = -1
	CallFunction(Player_SetHammerDashSpeed)
#endplatform
end function


function Player_State_HammerDash
#platform: Use_Origins
	if Player.Right == true
		Player.Direction = FACING_RIGHT
	else
		if Player.Left == true
			Player.Direction = FACING_LEFT
		end if
	end if

	TempValue0 = true

	if Player.Speed != 0
		Player.Timer++
		if Player.JumpHold == true
			if Player.Timer < 60
				if Player.Gravity == GRAVITY_AIR
					CallFunction(Player_HandleAirMovement)
				else
					CallFunction(Player_SetHammerDashSpeed)
					CallFunction(Player_HandleOnGround)
				end if
				TempValue0 = false
			end if
		end if
	end if

	// If Amy is on a steep enough slope, cancel the move
	// (Despite the patch notes for version 2.01 claiming this was removed, it's very much still here)
	Cos256(TempValue1, Player.Angle)
	if TempValue1 <= 0
		TempValue0 = true
	end if

	if TempValue0 == true
		Player.State = Player_State_Ground
		Player.Skidding = 0
	end if
#endplatform
end function

#platform: Use_Origins

		else
			Player.Speed = 0x60000
		end if
	else
		if Player.SuperState == SUPERSTATE_SUPER
			Player.Speed = -0xA0000
		else
			Player.Speed = -0x60000
		end if
	end if
#endplatform
end function

sub ObjectMain
	if Player.RestoreControl == true
		Player.ControlMode = CONTROLMODE_NORMAL
		Player.RestoreControl = false
	end if
	
	//if player.entityPos == SLOT_PLAYER1
		if Player.SuperState == SUPERSTATE_QUEUED
			Player.SuperState = SUPERSTATE_NONE
			CallFunction(Player_TryTransform)
		end if
	//end if
	
	CheckEqual(Player.State, Player_State_Spindash_S2)
	TempValue6 = checkResult
	CheckEqual(Player.State, Player_State_Spindash_CD)
	TempValue6 |= checkResult
	if TempValue6 == false
		SegaID.LoggedIn = 0
	end if

	// Only PC had access to debug mode, so here's a hack to prevent it on mobile
	TempValue0 = false
#platform: Standard
	TempValue0 = Stage.DebugMode
#endplatform

	if TempValue0 == true
		if Debug.S3kStyle == true
			TempValue0 = KeyPress[1].ButtonC
		else
			TempValue0 = KeyPress[1].ButtonB
		end if
		if TempValue0 == true	// Turn Debug Mode
			Object.Type = TypeName[Debug Mode]

			if Stage.PlayerListPos != PLAYER_SONIC_A // PLAYER_SONIC in origins
				Object[+1].Type = TypeName[Blank Object]
			end if

			Player.XVelocity	= 0
			Player.YVelocity	= 0
			Player.State		= Player_State_Static
			Player.Frame		= 0
			Player.Rotation		= 0

			Player.ObjectInteraction	= false
			Player.TileCollisions		= true
			Player.DrawOrder			= 4

			Player.AbilityTimer = 0
			Player.MinRollSpeed = 0

			Screen.CameraEnabled = true
			Screen.CameraStyle	 = Options.OriginalControls
		end if
	end if
	
	if Object.Type != TypeName[Debug Mode]
		// Mobile version ObjectMain starts here, since there wasn't a debug mode
		// Handle Player
		CallFunction(Player_ProcessUpdate)
		CallFunction(Player.State)

		ProcessAnimation()
#platform: Use_Origins
		if Player.Gravity == GRAVITY_GROUND
			Player.JumpAbilityState = 0 // If the player is on the ground, set the abilitystate to 0 so the player isn't able to do their air ability when rolling off a ledge.
		end if

		if Mini_PlayerFlag == false
			CheckEqual(Player.Animation, ANI_HAMMER_JUMP)
			TempValue5 = checkResult
			CheckEqual(Player.Animation, ANI_JUMPING)
			TempValue5 |= checkResult
			if TempValue5 == true
#endplatform
#platform: Use_Standalone
		if Mini_PlayerFlag == false
			if Player.Animation == ANI_JUMPING
#endplatform
				Screen.AdjustCameraY = Player.JumpOffset
			else
				if Screen.AdjustCameraY == Player.JumpOffset
					Screen.AdjustCameraY = 0
					Player.iYPos += Player.JumpOffset
				end if
			end if
		end if

		if Player.YVelocity > 0x100000
			Player.YVelocity = 0x100000
		end if

		TempValue6 = Player.PrevGravity
		Player.PrevGravity = Player.Gravity
		Player.PrevGravity ^= GRAVITY_AIR
		CheckEqual(Player.Gravity, GRAVITY_GROUND)
		Player.PrevGravity |= CheckResult
		Player.PrevGravity ^= GRAVITY_AIR
		if TempValue6 == GRAVITY_AIR
			if Player.PrevGravity == GRAVITY_GROUND
				Score.Timer = 0
			end if
		end if
		
		if Player.OffsetJump == false
			TempValue0 = Player.XPos
			TempValue1 = Player.YPos
			
			PlayerTileCollision()

			// if the player moved WAY too far after tile collisions were processed, they must've warped
			// so, put them back where they were
			
			TempValue2 = Player.XPos
			TempValue2 -= TempValue0
			if TempValue2 < 0
				FlipSign(TempValue2)
			end if
			if TempValue2 > 0x1000000
				Player.XPos = TempValue0
			end if
			
			TempValue2 = Player.YPos
			TempValue2 -= TempValue1
			if TempValue2 < 0
				FlipSign(TempValue2)
			end if
			if TempValue2 > 0x1000000
				Player.YPos = TempValue1
			end if
			
			if CheckResult == true
				if Player.Animation == ANI_JUMPING
					if Player.State != Player_State_BubbleBounce
						if Player.Down == false
							Player.Animation	 = ANI_WALKING
							Screen.AdjustCameraY = 0
							Player.iYPos		+= Player.JumpOffset
						end if
					end if
				end if
			end if
		end if
	end if
	
	CallFunction(Player_HandleSuperForm)
	CallFunction(Player_HandleHitboxOverride)
	
	if game.playMode == BOOT_PLAYMODE_BOSSRUSH
		if game.missionCondition == MISSION_CONDITION_CLEAR
			Player.InvincibleTimer = 80
			Player.AirTimer = 0
		end if
	end if
	
	Player.WasMainCalled = true
end sub


sub ObjectDraw
	// Now before you start yelling at me, hear me out...
	// Knuckles should enter an automatic glide drop if he glides into a wall's lower corner.
	// However, we can't start the drop until every stage object has run, as they can have climbable
	// collision of their own and might cancel out what was determined here in the player object.
	// But that means we need to run more player code AFTER everything else, but before the next frame,
	// and sticking it in the ObjectDraw event is the only way to do that...
	// WasMainCalled is used to keep this bit of code synchronized with ObjectMain, as ObjectDraw
	// sometimes runs when that doesn't, and we don't want that.
	TempValue7 = true
	if Player.WasMainCalled == true
		if Stage.PlayerListPos == PLAYER_KNUCKLES_A
			if Player.ShouldGlideDrop > 0
				Player.ShouldGlideDrop--
				if Player.ShouldGlideDrop == 0
					CheckEqual(Player.State, Player_State_GlideLeft)
					TempValue0 = CheckResult
					CheckEqual(Player.State, Player_State_GlideRight)
					TempValue0 |= CheckResult
					CheckEqual(Player.State, Player_State_GlideLeftNoGrip)
					TempValue0 |= CheckResult
					CheckEqual(Player.State, Player_State_GlideRightNoGrip)
					TempValue0 |= CheckResult
					
					if TempValue0 == true
						Player.XVelocity >>= 2
						Player.Speed = Player.XVelocity
						CallFunction(Player_Action_GlideDrop)
					end if
					
					Player.ShouldGlideDrop = 0
				end if
			else
				GetBit(TempValue0, Player.DelayLedgePull, 1)
				if TempValue0 == true
					SetBit(Player.DelayLedgePull, 1, false)
					if Player.State == Player_State_Climb	// make sure we even still want to ledge pull here
						Player.Animation = ANI_LEDGEPULLUP
						Player.YVelocity = 0
						Player.Timer = 0
						Player.State = Player_State_LedgePullUp
						
						Object.XPos = Player.XPos
						Object.YPos = Player.YPos
						if Player.Direction == FACING_RIGHT
							ObjectTileGrip(CSIDE_FLOOR, 12, -9, Player.CollisionPlane)
						else
							ObjectTileGrip(CSIDE_FLOOR, -12, -9, Player.CollisionPlane)
						end if
						if CheckResult == true
							Player.YPos = Object.YPos	// this was missing in vanilla Origins, resulting in the ObjectTileGrip doing NOTHING! :D
						end if
						
						Player.TileCollisions = false
						if Player.Direction == FACING_RIGHT
							Player.XPos += 0x10000
						else
							Player.XPos -= 0x10000
						end if
						
						TempValue7 = false
						
						// ...but it WORKS
						DrawSpriteFX(Mini_PlayerFlag, FX_FLIP, Player.XPos, Player.YPos)
					end if
				end if
			end if
		end if
	end if
	
	if Player.Animation != Player.PrevAnimation
		Player.PrevAnimation  = Player.Animation
		Player.Frame		  = 0
		Player.AnimationTimer = 0
		Player.AnimationSpeed = 0
	end if
	
	if TempValue7 == true
		TempValue1 = false
		if Stage.PlayerListPos == PLAYER_SONIC_A
			if Player.SuperState == SUPERSTATE_SUPER
				if Mini_PlayerFlag == false
					if Player.Animation == ANI_WALKING
						TempValue0 = Oscillation
						TempValue0 &= 3
						if TempValue0 == 0
							Player.Animation = ANI_PEELOUT
							TempValue1 = true
						end if
					end if
				end if
			end if
		end if
		
		DrawPlayerAnimation()
		
		if TempValue1 == true
			Player.Animation = ANI_WALKING
		end if
	end if
	
	Player.WasMainCalled = false
end sub

	// Origins Plus includes new start positions for Knuckles in some levels

	while ArrayPos1 < 1056
		// So now we have to check how many player objects the game found
		if Object[ArrayPos1].Type == TypeName[Player Object]
			TempValue5++
		end if
		ArrayPos1++
	loop

	// We found multiple spawn points, now check which one to use
	if TempValue5 > 1
		ArrayPos1  = 32
		TempValue7 = 0
		while ArrayPos1 < 1056
			if Object[ArrayPos1].Type == TypeName[Player Object]
				// The standard spawnpoint
				if Object[ArrayPos1].PropertyValue == 0
					TempValue7 = ArrayPos1
				end if

				// Character specific spawnpoint
				if Object[ArrayPos1].PropertyValue == Stage.PlayerListPos
					TempValue6 = ArrayPos1
					ArrayPos1 = 1056
				end if
			end if
			ArrayPos1++
		loop

		// If we didn't find a special spawnpoint, just use the regular one
		if TempValue6 == 0
			TempValue6 = TempValue7
		end if
	end if

	// Get rid of all spawnpoints except the one we're going to use
	TempValue7 = 0
	ArrayPos0 = 32
	while ArrayPos0 < 1056
		if Object[ArrayPos0].Type == TypeName[Player Object]
			if TempValue5 > 1
				if ArrayPos0 == TempValue6
					CallFunction(Player_Setup_Startup)
				else
					ResetObjectEntity(ArrayPos0, TypeName[Blank Object], 0, 0, 0)
				end if
			else
				TempValue7 = ArrayPos0
			end if
		end if
		ArrayPos0++
	loop

	if TempValue7 > 0
		ArrayPos0 = TempValue7
		CallFunction(Player_Setup_Startup)
	end if
	
	SegaID.LoggedIn = 0 // clear this on every startup so it's not an incorrect value somehow
#endplatform

#platform: Use_Standalone
	ArrayPos0 = 32
	while ArrayPos0 < 1056
		if Object[ArrayPos0].Type == TypeName[Player Object]
			switch Stage.PlayerListPos
			case PLAYER_SONIC_A
				ResetObjectEntity(0, TypeName[Player Object], 0, Object[ArrayPos0].XPos, Object[ArrayPos0].YPos)
				Player.XPos = Object[ArrayPos0].XPos
				Player.YPos = Object[ArrayPos0].YPos

				LoadAnimation("Sonic.Ani")
				BindPlayerToObject(0, 0)

				Player.State = Player_State_Air

				Player.Priority = PRIORITY_ACTIVE

				Player.DrawOrder = 4

				Player.GravityStrength		=  0x3800
				CallFunction(Player_UpdatePhysicsState)

				Player.JumpOffset			= -5

				Player.JumpAbility			= Player_State_Static
				if Options.OriginalControls == false
					Player.ActionPeelout = Player_Action_Peelout_S2
					Player.ActionSpindash = Player_Action_Spindash_S2
				else
					Player.ActionPeelout = Player_Action_Peelout_CD
					Player.ActionSpindash = Player_Action_Spindash_CD
				end if
				break

			case PLAYER_TAILS_A
				ResetObjectEntity(0, TypeName[Player Object], 0, Object[ArrayPos0].XPos, Object[ArrayPos0].YPos)
				Player.XPos = Object[ArrayPos0].XPos
				Player.YPos = Object[ArrayPos0].YPos

				LoadAnimation("Tails.Ani")
				BindPlayerToObject(0, 0)

				Player.State	 = Player_State_Air
				Player.Priority  = PRIORITY_ACTIVE
				Player.DrawOrder = 4

				Player.GravityStrength		=  0x3800
				CallFunction(Player_UpdatePhysicsState)

				Player.JumpOffset			= -1

				Player.JumpAbility			= Player_Action_DblJumpTails
				Player.ActionPeelout		= Player_Action_Jump
				if Options.OriginalControls == false
					Player.ActionSpindash = Player_Action_Spindash_S2
				else
					Player.ActionSpindash = Player_Action_Spindash_CD
				end if
				break

			end switch

			GetAnimationByName(ANI_SPINNING_TOP, "Spinning Top")
			GetAnimationByName(ANI_RAMP_RUNNING1, "3D Ramp 1")
			GetAnimationByName(ANI_RAMP_RUNNING2, "3D Ramp 2")
			GetAnimationByName(ANI_RAMP_RUNNING3, "3D Ramp 3")
			GetAnimationByName(ANI_RAMP_RUNNING4, "3D Ramp 4")
			GetAnimationByName(ANI_RAMP_RUNNING5, "3D Ramp 5")
			GetAnimationByName(ANI_RAMP_RUNNING6, "3D Ramp 6")
			GetAnimationByName(ANI_ROLL3D, "3D Ramp 7")
			GetAnimationByName(ANI_SIZE_CHANGE, "Size Change")

			ResetObjectEntity(ArrayPos0, TypeName[Blank Object], 0, 0, 0)
		end if

		ArrayPos0++
	loop
#endplatform

	Player.RestoreControl = false
	
	Player.JumpAbilityState = 0
	Player.DropDashCharge = -1
	
	Player.HitboxOverrideAnim	= -1
	Player.HitboxOverrideFrame	= -1
	
	// somehow this isn't always 0 by default in missions
	Player.Timer = 0
	
	//Stage.DebugMode = true
end sub

// All of the Character ID stuff is specific to Origins Plus, other versions of the game don't check or use the PropertyValue

sub RSDKEdit
	if Editor.ReturnVariable == true
		switch Editor.VariableID
		case EDIT_VAR_PROPVAL // Property Value
			CheckResult = Object.PropertyValue
			break
		case 0 // Character ID
			CheckResult  = Object.PropertyValue
			break
		end switch
	else
		switch Editor.VariableID
		case EDIT_VAR_PROPVAL // Property Value
			Object.PropertyValue = Editor.VariableValue
			break
		case 0 // Character ID
			Object.PropertyValue = Editor.VariableValue
			break
		end switch
	end if
end sub

	switch Object.PropertyValue

		break

	case 2

		DrawSprite(2)
		break

	case 5
		LoadSpriteSheet("Players/Amy1.gif")
		DrawSprite(3)
		break
	end switch
end sub
